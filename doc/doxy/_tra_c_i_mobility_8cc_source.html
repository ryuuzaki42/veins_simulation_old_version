146 void TraCIMobility::handleSelfMsg(cMessage *msg)
147 {
148  if (msg == startAccidentMsg) {
149  commandSetSpeed(0);
150  simtime_t accidentDuration = par("accidentDuration");
151  scheduleAt(simTime() + accidentDuration, stopAccidentMsg);

196 void TraCIMobility::changePosition()
197 {
198  // ensure we're not called twice in one time step
199  ASSERT(lastUpdate != simTime());
200 
201  // keep statistics (for current step)
202  currentPosXVec.record(move.getStartPos().x);
203  currentPosYVec.record(move.getStartPos().y);
204 
205  Coord nextPos = calculateAntennaPosition(roadPosition);
206  nextPos.z = move.getCurrentPosition().z;
207 
208  // keep statistics (relative to last step)
209  if (statistics.startTime != simTime()) {
210  simtime_t updateInterval = simTime() - this->lastUpdate;
211 
212  double distance = move.getStartPos().distance(nextPos);
213  statistics.totalDistance += distance;
214  statistics.totalTime += updateInterval;
215  if (speed != -1) {
216  statistics.minSpeed = std::min(statistics.minSpeed, speed);
217  statistics.maxSpeed = std::max(statistics.maxSpeed, speed);
218  currentSpeedVec.record(speed);
219  if (last_speed != -1) {
220  double acceleration = (speed - last_speed) / updateInterval;
221  double co2emission = calculateCO2emission(speed, acceleration);
222  currentAccelerationVec.record(acceleration);
223  currentCO2EmissionVec.record(co2emission);
224  statistics.totalCO2Emission+=co2emission * updateInterval.dbl();
225  }
226  last_speed = speed;
227  } else {
228  last_speed = -1;
229  speed = -1;
230  }
231  }
232  this->lastUpdate = simTime();
233 
234  move.setStart(Coord(nextPos.x, nextPos.y, move.getCurrentPosition().z)); // keep z position
235  move.setDirectionByVector(Coord(cos(angle), -sin(angle)));
236  move.setSpeed(speed);
237  if (ev.isGUI()) updateDisplayString();
238  fixIfHostGetsOutside();
239  updatePosition();
240 }

319 double TraCIMobility::calculateCO2emission(double v, double a) const {
320  // Calculate CO2 emission parameters according to:
321  // Cappiello, A. and Chabini, I. and Nam, E.K. and Lue, A. and Abou Zeid, M., "A statistical model of vehicle emissions and fuel consumption," IEEE 5th International Conference on Intelligent Transportation Systems (IEEE ITSC), pp. 801-809, 2002
322 
323  double A = 1000 * 0.1326; // W/m/s
324  double B = 1000 * 2.7384e-03; // W/(m/s)^2
325  double C = 1000 * 1.0843e-03; // W/(m/s)^3
326  double M = 1325.0; // kg
327 
328  // power in W
329  double P_tract = A*v + B*v*v + C*v*v*v + M*a*v; // for sloped roads: +M*g*sin_theta*v
330 
331  /*
332  // "Category 7 vehicle" (e.g. a '92 Suzuki Swift)
333  double alpha = 1.01;
334  double beta = 0.0162;
335  double delta = 1.90e-06;
336  double zeta = 0.252;
337  double alpha1 = 0.985;
338  */
339 
340  // "Category 9 vehicle" (e.g. a '94 Dodge Spirit)
341  double alpha = 1.11;
342  double beta = 0.0134;
343  double delta = 1.98e-06;
344  double zeta = 0.241;
345  double alpha1 = 0.973;
346 
347  if (P_tract <= 0) return alpha1;
348  return alpha + beta*v*3.6 + delta*v*v*v*(3.6*3.6*3.6) + zeta*a*v;
349 }
