52 void BaseMacLayer::initialize(int stage)
53 {
54  BaseLayer::initialize(stage);
55 
56  if(stage==0)
57  {
58  // get handle to phy layer
59  if ((phy = FindModule<MacToPhyInterface*>::findSubModule(getParentModule())) == NULL) {
60  error("Could not find a PHY module.");
61  }
62  headerLength = par("headerLength");
63  phyHeaderLength = phy->getPhyHeaderLength();
64 
65  hasPar("coreDebug") ? coreDebug = par("coreDebug").boolValue() : coreDebug = false;
66  }
67  if (myMacAddr == LAddress::L2NULL) {
68  // see if there is an addressing module available
69  // otherwise use NIC modules id as MAC address
70  AddressingInterface* addrScheme = FindModule<AddressingInterface*>::findSubModule(findHost());
71  if(addrScheme) {
72  myMacAddr = addrScheme->myMacAddr(this);
73  } else {
74  const std::string addressString = par("address").stringValue();
75  if (addressString.empty() || addressString == "auto")
76  myMacAddr = LAddress::L2Type(getParentModule()->getId());
77  else
78  myMacAddr = LAddress::L2Type(addressString.c_str());
79  // use streaming operator for string conversion, this makes it more
80  // independent from the myMacAddr type
81  std::ostringstream oSS; oSS << myMacAddr;
82  par("address").setStringValue(oSS.str());
83  }
84  registerInterface();
85  }
86 }

88 void BaseMacLayer::registerInterface()
89 {
90 #ifdef MIXIM_INET
91  IInterfaceTable *ift = InterfaceTableAccess().getIfExists();
92  if (ift) {
93  cModule* nic = getParentModule();
94  InterfaceEntry *e = new InterfaceEntry();
95 
96  // interface name: NIC module's name without special
97  // characters ([])
98  char *interfaceName = new char[strlen(nic->getFullName()) + 1];
99  char *d = interfaceName;
100  for (const char *s = nic->getFullName(); *s; s++)
101  if (isalnum(*s))
102  *d++ = *s;
103  *d = '\0';
104 
105  e->setName(interfaceName);
106  delete [] interfaceName;
107 
108  // this MAC address must be the same as the one in BaseMacLayer
109  e->setMACAddress(myMacAddr);
110 
111  // generate interface identifier for IPv6
112  e->setInterfaceToken(myMacAddr.formInterfaceIdentifier());
113 
114  // MTU on 802.11 = ?
115  e->setMtu(1500); // FIXME
116 
117  // capabilities
118  e->setBroadcast(true);
119  e->setMulticast(true);
120  e->setPointToPoint(false);
121 
122  // add
123  ift->addInterface(e, this);
124  }
125 #endif

191 void BaseMacLayer::handleLowerMsg(cMessage *msg)
192 {
193  MacPkt* mac = static_cast<MacPkt *>(msg);
194  LAddress::L2Type dest = mac->getDestAddr();
195  LAddress::L2Type src = mac->getSrcAddr();
196 
197  //only foward to upper layer if message is for me or broadcast
198  if((dest == myMacAddr) || LAddress::isL2Broadcast(dest)) {
199  coreEV << "message with mac addr " << src
200  << " for me (dest=" << dest
201  << ") -> forward packet to upper layer\n";
202  sendUp(decapsMsg(mac));
203  }
204  else{
205  coreEV << "message with mac addr " << src
206  << " not for me (dest=" << dest
207  << ") -> delete (my MAC="<<myMacAddr<<")\n";
208  delete mac;
209  }
210 }
