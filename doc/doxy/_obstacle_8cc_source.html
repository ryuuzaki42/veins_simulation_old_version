59  bool isPointInObstacle(Coord point, const Obstacle& o) {
60  bool isInside = false;
61  const Obstacle::Coords& shape = o.getShape();
62  Obstacle::Coords::const_iterator i = shape.begin();
63  Obstacle::Coords::const_iterator j = (shape.rbegin()+1).base();
64  for (; i != shape.end(); j = i++) {
65  bool inYRangeUp = (point.y >= i->y) && (point.y < j->y);
66  bool inYRangeDown = (point.y >= j->y) && (point.y < i->y);
67  bool inYRange = inYRangeUp || inYRangeDown;
68  if (!inYRange) continue;
69  bool intersects = point.x < (i->x + ((point.y - i->y) * (j->x - i->x) / (j->y - i->y)));
70  if (!intersects) continue;
71  isInside = !isInside;
72  }
73  return isInside;
74  }

76  double segmentsIntersectAt(Coord p1From, Coord p1To, Coord p2From, Coord p2To) {
77  Coord p1Vec = p1To - p1From;
78  Coord p2Vec = p2To - p2From;
79  Coord p1p2 = p1From - p2From;
80 
81  double D = (p1Vec.x * p2Vec.y - p1Vec.y * p2Vec.x);
82 
83  double p1Frac = (p2Vec.x * p1p2.y - p2Vec.y * p1p2.x) / D;
84  if (p1Frac < 0 || p1Frac > 1) return -1;
85 
86  double p2Frac = (p1Vec.x * p1p2.y - p1Vec.y * p1p2.x) / D;
87  if (p2Frac < 0 || p2Frac > 1) return -1;
88 
89  return p1Frac;
90  }

93 double Obstacle::calculateAttenuation(const Coord& senderPos, const Coord& receiverPos) const {
94 
95  // if obstacles has neither walls nor matter: bail.
96  if (getShape().size() < 2) return 1;
97 
98  // get a list of points (in [0, 1]) along the line between sender and receiver where the beam intersects with this obstacle
99  std::multiset<double> intersectAt;
100  bool doesIntersect = false;
101  const Obstacle::Coords& shape = getShape();
102  Obstacle::Coords::const_iterator i = shape.begin();
103  Obstacle::Coords::const_iterator j = (shape.rbegin()+1).base();
104  for (; i != shape.end(); j = i++) {
105  Coord c1 = *i;
106  Coord c2 = *j;
107 
108  double i = segmentsIntersectAt(senderPos, receiverPos, c1, c2);
109  if (i != -1) {
110  doesIntersect = true;
111  intersectAt.insert(i);
112  }
113 
114  }
115 
116  // if beam interacts with neither walls nor matter: bail.
117  bool senderInside = isPointInObstacle(senderPos, *this);
118  bool receiverInside = isPointInObstacle(receiverPos, *this);
119  if (!doesIntersect && !senderInside && !receiverInside) return 1;
120 
121  // remember number of walls before messing with intersection points
122  double numWalls = intersectAt.size();
123 
124  // for distance calculation, make sure every other pair of points marks transition through matter and void, respectively.
125  if (senderInside) intersectAt.insert(0);
126  if (receiverInside) intersectAt.insert(1);
127  ASSERT((intersectAt.size() % 2) == 0);
128 
129  // sum up distances in matter.
130  double fractionInObstacle = 0;
131  for (std::multiset<double>::const_iterator i = intersectAt.begin(); i != intersectAt.end(); ) {
132  double p1 = *(i++);
133  double p2 = *(i++);
134  fractionInObstacle += (p2 - p1);
135  }
136 
137  // calculate attenuation
138  double totalDistance = senderPos.distance(receiverPos);
139  double attenuation = (attenuationPerWall * numWalls) + (attenuationPerMeter * fractionInObstacle * totalDistance);
140  return pow(10.0, -attenuation/10.0);
141 }
