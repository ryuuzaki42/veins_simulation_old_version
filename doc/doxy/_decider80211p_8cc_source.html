36 simtime_t Decider80211p::processNewSignal(AirFrame* msg) {
37 
38  AirFrame11p *frame = check_and_cast<AirFrame11p *>(msg);
39 
40  // get the receiving power of the Signal at start-time and center frequency
41  Signal& signal = frame->getSignal();
42 
43  Argument start(DimensionSet::timeFreqDomain);
44  start.setTime(signal.getReceptionStart());
45  start.setArgValue(Dimension::frequency_static(), centerFrequency);
46 
47  signalStates[frame] = EXPECT_END;
48 
49  double recvPower = signal.getReceivingPower()->getValue(start);
50 
51  if (recvPower < sensitivity) {
52  //annotate the frame, so that we won't try decoding it at its end
53  frame->setUnderSensitivity(true);
54  //check channel busy status. a superposition of low power frames might turn channel status to busy
55  if (cca(simTime(), NULL) == false) {
56  setChannelIdleStatus(false);
57  }
58  return signal.getReceptionEnd();
59  }
60  else {
61 
62  setChannelIdleStatus(false);
63 
64  if (phy11p->getRadioState() == Radio::TX) {
65  frame->setBitError(true);
66  frame->setWasTransmitting(true);
67  DBG_D11P << "AirFrame: " << frame->getId() << " (" << recvPower << ") received, while already sending. Setting BitErrors to true" << std::endl;
68  }
69  else {
70 
71  if (!curSyncFrame) {
72  //NIC is not yet synced to any frame, so lock and try to decode this frame
73  curSyncFrame = frame;
74  DBG_D11P << "AirFrame: " << frame->getId() << " with (" << recvPower << " > " << sensitivity << ") -> Trying to receive AirFrame." << std::endl;
75  }
76  else {
77  //NIC is currently trying to decode another frame. this frame will be simply treated as interference
78  DBG_D11P << "AirFrame: " << frame->getId() << " with (" << recvPower << " > " << sensitivity << ") -> Already synced to another AirFrame. Treating AirFrame as interference." << std::endl;
79  }
80 
81 
82  //channel turned busy
83  //measure communication density
84  myBusyTime += signal.getDuration().dbl();
85  }
86  return signal.getReceptionEnd();
87  }

121  Signal& signal = frame->getSignal();
122 
123  simtime_t start = signal.getReceptionStart();
124  simtime_t end = signal.getReceptionEnd();

126  //call BaseDecider function to get Noise plus Interference mapping

147 Mapping* Decider80211p::calculateNoiseRSSIMapping(simtime_t_cref start, simtime_t_cref end, AirFrame *exclude) {
148 
149  // create an empty mapping
150  Mapping* resultMap = MappingUtils::createMapping(Argument::MappedZero, DimensionSet::timeDomain);
151 
152  // add thermal noise
153  ConstMapping* thermalNoise = phy->getThermalNoise(start, end);
154  if (thermalNoise) {
155  // FIXME: workaround needed to make *really* sure that the resultMap is defined for the range of the exclude-frame
156  const ConstMapping* excludePwr = exclude ? exclude->getSignal().getReceivingPower() : 0;
157  if (excludePwr) {
158  Mapping* p1 = resultMap;
159  // p2 = exclude + thermal
160  Mapping* p2 = MappingUtils::add(*excludePwr, *thermalNoise);
161  // p3 = p2 - exclude
162  Mapping* p3 = MappingUtils::subtract(*p2, *excludePwr);
163  // result = result + p3
164  resultMap = MappingUtils::add(*resultMap, *p3);
165  delete p3;
166  delete p2;
167  delete p1;
168  }
169  else {
170  Mapping* p1 = resultMap;
171  resultMap = MappingUtils::add(*resultMap, *thermalNoise);
172  delete p1;
173  }
174  }
175 
176  return resultMap;

180 DeciderResult* Decider80211p::checkIfSignalOk(AirFrame* frame) {
181 
182  Mapping* sinrMap = 0;
183  Mapping *snrMap = 0;
184 
185  if (collectCollisionStats) {
186  calculateSinrAndSnrMapping(frame, &sinrMap, &snrMap);
187  assert(snrMap);
188  }
189  else {
190  sinrMap = calculateSnrMapping(frame);
191  }
192  assert(sinrMap);
193 
194  Signal& s = frame->getSignal();
195  simtime_t start = s.getReceptionStart();
196  simtime_t end = s.getReceptionEnd();
197 
198  start = start + PHY_HDR_PREAMBLE_DURATION; //its ok if something in the training phase is broken
199 
200  Argument min(DimensionSet::timeFreqDomain);
201  min.setTime(start);
202  min.setArgValue(Dimension::frequency_static(), centerFrequency - 5e6);
203  Argument max(DimensionSet::timeFreqDomain);
204  max.setTime(end);
205  max.setArgValue(Dimension::frequency_static(), centerFrequency + 5e6);
206 
207  double snirMin = MappingUtils::findMin(*sinrMap, min, max);
208  double snrMin;
209  if (collectCollisionStats) {
210  snrMin = MappingUtils::findMin(*snrMap, min, max);
211  }
212  else {
213  //just set to any value. if collectCollisionStats != true
214  //it will be ignored by packetOk
215  snrMin = 1e200;
216  }
217 
218  ConstMappingIterator* bitrateIt = s.getBitrate()->createConstIterator();
219  bitrateIt->next(); //iterate to payload bitrate indicator
220  double payloadBitrate = bitrateIt->getValue();
221  delete bitrateIt;
222 
223  DeciderResult80211* result = 0;
224 
225  switch (packetOk(snirMin, snrMin, frame->getBitLength(), payloadBitrate)) {
226 
227  case DECODED:
228  DBG_D11P << "Packet is fine! We can decode it" << std::endl;
229  result = new DeciderResult80211(true, payloadBitrate, snirMin);
230  break;
231 
232  case NOT_DECODED:
233  if (!collectCollisionStats) {
234  DBG_D11P << "Packet has bit Errors. Lost " << std::endl;
235  }
236  else {
237  DBG_D11P << "Packet has bit Errors due to low power. Lost " << std::endl;
238  }
239  result = new DeciderResult80211(false, payloadBitrate, snirMin);
240  break;
241 
242  case COLLISION:
243  DBG_D11P << "Packet has bit Errors due to collision. Lost " << std::endl;
244  collisions++;
245  result = new DeciderResult80211(false, payloadBitrate, snirMin);
246  break;
247 
248  default:
249  ASSERT2(false, "Impossible packet result returned by packetOk(). Check the code.");
250  break;
251 
252  }
253 
254  delete sinrMap;
255  if (snrMap)
256  delete snrMap;
257  return result;
258 }

260 enum Decider80211p::PACKET_OK_RESULT Decider80211p::packetOk(double snirMin, double snrMin, int lengthMPDU, double bitrate) {
261 
262  //the lengthMPDU includes the PHY_SIGNAL_LENGTH + PHY_PSDU_HEADER + Payload, while the first is sent with PHY_HEADER_BANDWIDTH
263 
264  double packetOkSinr;
265  double packetOkSnr;
266 
267  if (bitrate == 18E+6) {
268  //According to P. Fuxjaeger et al. "IEEE 802.11p Transmission Using GNURadio"
269  double ber = std::min(0.5 , 1.5 * erfc(0.45 * sqrt(snirMin)));
270  packetOkSinr = pow(1 - ber, lengthMPDU - PHY_HDR_PLCPSIGNAL_LENGTH);
271  }
272  else if (bitrate == 6E+6) {
273  //According to K. Sjoeberg et al. "Measuring and Using the RSSI of IEEE 802.11p"
274  double ber = std::min(0.5 , 8 * erfc(0.75 *sqrt(snirMin)));
275  packetOkSinr = pow(1 - ber, lengthMPDU - PHY_HDR_PLCPSIGNAL_LENGTH);
276  }
277  else {
278  opp_error("Currently this 11p-Model only provides accurate BER models for 6Mbit and 18Mbit. Please use one of these frequencies for now.");
279  }
280 
281  //check if header is broken, BER model for PSK taken from MiXiM 2.2
282  double berHeader = 0.5 * exp(-snirMin * 10E+6 / PHY_HDR_BANDWIDTH);
283  double headerNoError = pow(1.0 - berHeader, PHY_HDR_PLCPSIGNAL_LENGTH);
284 
285  double headerNoErrorSnr;
286  //compute PER also for SNR only
287  if (collectCollisionStats) {
288  if (bitrate == 18E+6) {
289  //According to P. Fuxjaeger et al. "IEEE 802.11p Transmission Using GNURadio"
290  double ber = std::min(0.5, 1.5 * erfc(0.45 * sqrt(snrMin)));
291  packetOkSnr = pow(1 - ber, lengthMPDU - PHY_HDR_PLCPSIGNAL_LENGTH);
292  }
293  else if (bitrate == 6E+6) {
294  //According to K. Sjoeberg et al. "Measuring and Using the RSSI of IEEE 802.11p"
295  double ber = std::min(0.5, 8 * erfc(0.75 * sqrt(snrMin)));
296  packetOkSnr = pow(1 - ber, lengthMPDU - PHY_HDR_PLCPSIGNAL_LENGTH);
297  }
298 
299  double berHeader = 0.5 * exp(-snrMin * 10E+6 / PHY_HDR_BANDWIDTH);
300  headerNoErrorSnr = pow(1.0 - berHeader, PHY_HDR_PLCPSIGNAL_LENGTH);
301 
302  //the probability of correct reception without considering the interference
303  //MUST be greater or equal than when consider it
304  assert(packetOkSnr >= packetOkSinr);
305  assert(headerNoErrorSnr >= headerNoError);
306 
307  }
308 
309  //probability of no bit error in the PLCP header
310 
311  double rand = dblrand();
312 
313  if (!collectCollisionStats) {
314  if (rand > headerNoError)
315  return NOT_DECODED;
316  }
317  else {
318 
319  if (rand > headerNoError) {
320  //ups, we have a header error. is that due to interference?
321  if (rand > headerNoErrorSnr) {
322  //no. we would have not been able to receive that even
323  //without interference
324  return NOT_DECODED;
325  }
326  else {
327  //yes. we would have decoded that without interference
328  return COLLISION;
329  }
330 
331  }
332 
333  }
334 
335  //probability of no bit error in the rest of the packet
336 
337  rand = dblrand();
338 
339  if (!collectCollisionStats) {
340  if (rand > packetOkSinr) {
341  return NOT_DECODED;
342  }
343  else {
344  return DECODED;
345  }
346  }
347  else {
348 
349  if (rand > packetOkSinr) {
350  //ups, we have an error in the payload. is that due to interference?
351  if (rand > packetOkSnr) {
352  //no. we would have not been able to receive that even
353  //without interference
354  return NOT_DECODED;
355  }
356  else {
357  //yes. we would have decoded that without interference
358  return COLLISION;
359  }
360 
361  }
362  else {
363  return DECODED;
364  }
365 
366  }
367 }

370 bool Decider80211p::cca(simtime_t_cref time, AirFrame* exclude) {
371  AirFrameVector airFrames;
372 
373  // collect all AirFrames that intersect with [start, end]
374  getChannelInfo(time, time, airFrames);
375 
376  Mapping* resultMap = MappingUtils::createMapping(Argument::MappedZero, DimensionSet::timeDomain);
377 
378 
379  // iterate over all AirFrames (except exclude)
380  // and sum up their receiving-power-mappings
381  for (AirFrameVector::const_iterator it = airFrames.begin(); it != airFrames.end(); ++it) {
382  if (*it == exclude) {
383  continue;
384  }
385  // the vector should not contain pointers to 0
386  assert(*it != 0);
387 
388  // if iterator points to exclude (that includes the default-case 'exclude == 0')
389  // then skip this AirFrame
390 
391  // otherwise get the Signal and its receiving-power-mapping
392  Signal& signal = (*it)->getSignal();
393 
394  // backup pointer to result map
395  // Mapping* resultMapOld = resultMap;
396 
397  // TODO1.1: for testing purposes, for now we don't specify an interval
398  // and add the Signal's receiving-power-mapping to resultMap in [start, end],
399  // the operation Mapping::add returns a pointer to a new Mapping
400 
401  const ConstMapping* const recvPowerMap = signal.getReceivingPower();
402  assert(recvPowerMap);
403 
404  // Mapping* resultMapNew = Mapping::add( *(signal.getReceivingPower()), *resultMap, start, end );
405 
406  Mapping* resultMapNew = MappingUtils::add(*recvPowerMap, *resultMap, Argument::MappedZero);
407 
408  // discard old mapping
409  delete resultMap;
410  resultMap = resultMapNew;
411  resultMapNew = 0;
412  }
413 
414  //add thermal noise
415  ConstMapping* thermalNoise = phy->getThermalNoise(time, time);
416  if (thermalNoise) {
417  Mapping* tmp = resultMap;
418  resultMap = MappingUtils::add(*resultMap, *thermalNoise);
419  delete tmp;
420  }
421 
422  Argument min(DimensionSet::timeFreqDomain);
423  min.setTime(time);
424  min.setArgValue(Dimension::frequency_static(), centerFrequency - 5e6);
425 
426  DBG_D11P << MappingUtils::findMin(*resultMap, min, min) << " > " << sensitivity << " = " << (bool)(MappingUtils::findMin(*resultMap, min, min) > sensitivity) << std::endl;
427  bool isChannelIdle = MappingUtils::findMin(*resultMap, min, min) < sensitivity;
428  delete resultMap;
429  return isChannelIdle;
430 }

433 simtime_t Decider80211p::processSignalEnd(AirFrame* msg) {
434 
435  AirFrame11p *frame = check_and_cast<AirFrame11p *>(msg);
436 
437  // here the Signal is finally processed
438 
439  bool whileSending = false;
440 
441  //remove this frame from our current signals
442  signalStates.erase(frame);
443 
444  DeciderResult* result;
445 
446  if (frame->getUnderSensitivity()) {
447  //this frame was not even detected by the radio card
448  result = new DeciderResult80211(false,0,0);
449  }
450  else if (frame->getWasTransmitting() || phy11p->getRadioState() == Radio::TX) {
451  //this frame was received while sending
452  whileSending = true;
453  result = new DeciderResult80211(false,0,0);
454  }
455  else {
456 
457  //first check whether this is the frame NIC is currently synced on
458  if (frame == curSyncFrame) {
459  // check if the snrMapping is above the Decider's specific threshold,
460  // i.e. the Decider has received it correctly
461  result = checkIfSignalOk(frame);
462 
463  //after having tried to decode the frame, the NIC is no more synced to the frame
464  //and it is ready for syncing on a new one
465  curSyncFrame = 0;
466  }
467  else {
468  //if this is not the frame we are synced on, we cannot receive it
469  result = new DeciderResult80211(false, 0, 0);
470  }
471  }
472 
473  if (result->isSignalCorrect()) {
474  DBG_D11P << "packet was received correctly, it is now handed to upper layer...\n";
475  // go on with processing this AirFrame, send it to the Mac-Layer
476  phy->sendUp(frame, result);
477  }
478  else {
479  if (frame->getUnderSensitivity()) {
480  DBG_D11P << "packet was not detected by the card. power was under sensitivity threshold\n";
481  }
482  else if (whileSending) {
483  DBG_D11P << "packet was received while sending, sending it as control message to upper layer\n";
484  phy->sendControlMsgToMac(new cMessage("Error",RECWHILESEND));
485  }
486  else {
487  DBG_D11P << "packet was not received correctly, sending it as control message to upper layer\n";
488  phy->sendControlMsgToMac(new cMessage("Error",BITERROR));
489  }
490  delete result;
491  }
492 
493  if (phy11p->getRadioState() == Radio::TX) {
494  DBG_D11P << "I'm currently sending\n";
495  }
496  //check if channel is idle now
497  else if (cca(simTime(), frame) == false) {
498  DBG_D11P << "Channel not yet idle!\n";
499  }
500  else {
501  //might have been idle before (when the packet rxpower was below sens)
502  if (isChannelIdle != true) {
503  DBG_D11P << "Channel idle now!\n";
504  setChannelIdleStatus(true);
505  }
506  }
507  return notAgain;
508 }
