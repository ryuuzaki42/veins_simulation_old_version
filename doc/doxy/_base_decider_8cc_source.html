33 simtime_t BaseDecider::processNewSignal(AirFrame* frame) {
34  if(currentSignal.first != 0) {
35  deciderEV << "Already receiving another AirFrame!" << endl;
36  return notAgain;
37  }
38 
39  // get the receiving power of the Signal at start-time
40  Signal& signal = frame->getSignal();
41  double recvPower = signal.getReceivingPower()->getValue(Argument(signal.getReceptionStart()));
42 
43  // check whether signal is strong enough to receive
44  if ( recvPower < sensitivity )
45  {
46  deciderEV << "Signal is to weak (" << recvPower << " < " << sensitivity
47  << ") -> do not receive." << endl;
48  // Signal too weak, we can't receive it, tell PhyLayer that we don't want it again
49  return notAgain;
50  }
51 
52  // Signal is strong enough, receive this Signal and schedule it
53  deciderEV << "Signal is strong enough (" << recvPower << " > " << sensitivity
54  << ") -> Trying to receive AirFrame." << endl;
55 
56  currentSignal.first = frame;
57  currentSignal.second = EXPECT_END;
58 
59  //channel turned busy
60  setChannelIdleStatus(false);
61 
62  return signal.getReceptionEnd();
63 }

113 simtime_t BaseDecider::handleNewSenseRequest(ChannelSenseRequest* request)
114 {
115  // no request handled at the moment, handling the new one
116  simtime_t now = phy->getSimTime();
117 
118  // saving the pointer to the request and its start-time (now)
119  currentChannelSenseRequest.setRequest(request);
120  currentChannelSenseRequest.setSenseStart(now);
121 
122  //get point in time when we can answer the request (as far as we
123  //know at this point in time)
124  currentChannelSenseRequest.canAnswerAt = canAnswerCSR(currentChannelSenseRequest);
125 
126  //check if we can already answer the request
127  if(now == currentChannelSenseRequest.canAnswerAt)
128  {
129  answerCSR(currentChannelSenseRequest);
130  return notAgain;
131  }
132 
133  return currentChannelSenseRequest.canAnswerAt;

148 void BaseDecider::channelStateChanged()
149 {
150  if(!currentChannelSenseRequest.getRequest())
151  return;
152 
153  //check if the point in time when we can answer the request has changed
154  simtime_t canAnswerAt = canAnswerCSR(currentChannelSenseRequest);
155 
156  //check if answer time has changed
157  if(canAnswerAt != currentChannelSenseRequest.canAnswerAt) {
158  //can we answer it now?
159  if(canAnswerAt == phy->getSimTime()) {
160  phy->cancelScheduledMessage(currentChannelSenseRequest.getRequest());
161  answerCSR(currentChannelSenseRequest);
162  } else {
163  phy->rescheduleMessage(currentChannelSenseRequest.getRequest(),
164  canAnswerAt);
165  currentChannelSenseRequest.canAnswerAt = canAnswerAt;
166  }
167  }
