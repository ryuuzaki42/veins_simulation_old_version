41 void TraCIScenarioManagerLaunchd::initialize(int stage)
42 {
43  if (stage != 1) {
44  TraCIScenarioManager::initialize(stage);
45  return;
46  }
47  launchConfig = par("launchConfig").xmlValue();
48  seed = par("seed");
49  cXMLElementList basedir_nodes = launchConfig->getElementsByTagName("basedir");
50  if (basedir_nodes.size() == 0) {
51  // default basedir is where current network file was loaded from
52  std::string basedir = cSimulation::getActiveSimulation()->getEnvir()->getConfig()->getConfigEntry("network").getBaseDirectory();
53  cXMLElement* basedir_node = new cXMLElement("basedir", __FILE__, launchConfig);
54  basedir_node->setAttribute("path", basedir.c_str());
55  launchConfig->appendChild(basedir_node);
56  }
57  cXMLElementList seed_nodes = launchConfig->getElementsByTagName("seed");
58  if (seed_nodes.size() == 0) {
59  if (seed == -1) {
60  // default seed is current repetition
61  const char* seed_s = cSimulation::getActiveSimulation()->getEnvir()->getConfigEx()->getVariable(CFGVAR_RUNNUMBER);
62  seed = atoi(seed_s);
63  }
64  std::stringstream ss; ss << seed;
65  cXMLElement* seed_node = new cXMLElement("seed", __FILE__, launchConfig);
66  seed_node->setAttribute("value", ss.str().c_str());
67  launchConfig->appendChild(seed_node);
68  }
69  TraCIScenarioManager::initialize(stage);
70 }

77 void TraCIScenarioManagerLaunchd::init_traci() {
78  {
79  std::pair<uint32_t, std::string> version = getCommandInterface()->getVersion();
80  uint32_t apiVersion = version.first;
81  std::string serverVersion = version.second;
82 
83  if ((apiVersion == 1)) {
84  MYDEBUG << "TraCI server \"" << serverVersion << "\" reports API version " << apiVersion << endl;
85  }
86  else {
87  error("TraCI server \"%s\" reports API version %d, which is unsupported. We recommend using the version of sumo-launchd that ships with Veins.", serverVersion.c_str(), apiVersion);
88  }
89  }
90 
91  std::string contents = launchConfig->tostr(0);
92 
93  TraCIBuffer buf;
94  buf << std::string("sumo-launchd.launch.xml") << contents;
95  connection->sendMessage(makeTraCICommand(CMD_FILE_SEND, buf));
96 
97  TraCIBuffer obuf(connection->receiveMessage());
98  uint8_t cmdLength; obuf >> cmdLength;
99  uint8_t commandResp; obuf >> commandResp; if (commandResp != CMD_FILE_SEND) error("Expected response to command %d, but got one for command %d", CMD_FILE_SEND, commandResp);
100  uint8_t result; obuf >> result;
101  std::string description; obuf >> description;
102  if (result != RTYPE_OK) {
103  EV << "Warning: Received non-OK response from TraCI server to command " << CMD_FILE_SEND << ":" << description.c_str() << std::endl;
104  }
105 
106  TraCIScenarioManager::init_traci();
107 }
