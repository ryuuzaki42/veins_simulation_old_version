39 void osdp_RSU::handleSelfMsg(cMessage* msg) {
40  //Code to realize by the RSU when it receives scheduled events
41  switch (msg->getKind()) {
42  case SEND_BEACON_EVT: {
43  sendWSM(prepareWSM("beacon", beaconLengthBits, type_CCH, beaconPriority, 0, -1));
44  scheduleAt(simTime() + par("beaconInterval").doubleValue(), sendBeaconEvt);

48  case SERVICE_EVT: {
49  for (int i=serv1-1; i< serv2; i++) {
