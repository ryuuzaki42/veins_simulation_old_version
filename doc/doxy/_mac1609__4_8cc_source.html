136 void Mac1609_4::handleSelfMsg(cMessage* msg) {
137  if (msg == nextChannelSwitch) {
138  ASSERT(useSCH);
139 
140  scheduleAt(simTime() + SWITCHING_INTERVAL_11P, nextChannelSwitch);
141 
142  switch (activeChannel) {
143  case type_CCH:
144  DBG_MAC << "CCH --> SCH" << std::endl;
145  channelBusySelf(false);
146  setActiveChannel(type_SCH);
147  channelIdle(true);
148  phy11p->changeListeningFrequency(frequency[mySCH]);
149  break;
150  case type_SCH:
151  DBG_MAC << "SCH --> CCH" << std::endl;
152  channelBusySelf(false);
153  setActiveChannel(type_CCH);
154  channelIdle(true);
155  phy11p->changeListeningFrequency(frequency[Channels::CCH]);
156  break;
157  }
158  //schedule next channel switch in 50ms
159 
160  }
161  else if (msg == nextMacEvent) {
162 
163  //we actually came to the point where we can send a packet
164  channelBusySelf(true);
165  WaveShortMessage* pktToSend = myEDCA[activeChannel]->initiateTransmit(lastIdle);
166 
167  lastAC = mapPriority(pktToSend->getPriority());
168 
169  DBG_MAC << "MacEvent received. Trying to send packet with priority" << lastAC << std::endl;
170 
171  //send the packet
172  Mac80211Pkt* mac = new Mac80211Pkt(pktToSend->getName(), pktToSend->getKind());
173  mac->setDestAddr(LAddress::L2BROADCAST);
174  mac->setSrcAddr(myMacAddress);
175  mac->encapsulate(pktToSend->dup());
176 
177  simtime_t sendingDuration = RADIODELAY_11P + getFrameDuration(mac->getBitLength());
178  DBG_MAC << "Sending duration will be" << sendingDuration << std::endl;
179  if ((!useSCH) || (timeLeftInSlot() > sendingDuration)) {
180  if (useSCH) DBG_MAC << " Time in this slot left: " << timeLeftInSlot() << std::endl;
181  // give time for the radio to be in Tx state before transmitting
182  phy->setRadioState(Radio::TX);
183 
184 
185  double freq = (activeChannel == type_CCH) ? frequency[Channels::CCH] : frequency[mySCH];
186 
187  attachSignal(mac, simTime()+RADIODELAY_11P, freq);
188  MacToPhyControlInfo* phyInfo = dynamic_cast<MacToPhyControlInfo*>(mac->getControlInfo());
189  assert(phyInfo);
190  DBG_MAC << "Sending a Packet. Frequency " << freq << " Priority" << lastAC << std::endl;
191  sendDelayed(mac, RADIODELAY_11P, lowerLayerOut);
192  statsSentPackets++;
193  }
194  else { //not enough time left now
195  DBG_MAC << "Too little Time left. This packet cannot be send in this slot." << std::endl;
196  statsNumTooLittleTime++;
197  //revoke TXOP
198  myEDCA[activeChannel]->revokeTxOPs();
199  delete mac;
200  channelIdle();
201  //do nothing. contention will automatically start after channel switch
202  }
203  }

211

213  if ((thisMsg = dynamic_cast<WaveShortMessage*>(msg)) == NULL) {
214  error("WaveMac only accepts WaveShortMessages");
215  }
216 
217  t_access_category ac = mapPriority(thisMsg->getPriority());

221  << ac << std::endl;

224

246  //if this packet is not at the front of a new queue we dont have to reevaluate times

248 
249  if (chan == activeChannel) {
250  DBG_MAC << "this packet is for the currently active channel" << std::endl;
251  }
252  else {
253  DBG_MAC << "this packet is NOT for the currently active channel" << std::endl;
254  }
255 
256  if (num == 1 && idleChannel == true && chan == activeChannel) {
257 
258  simtime_t nextEvent = myEDCA[chan]->startContent(lastIdle,guardActive());
259 
260  if (nextEvent != -1) {
261  if ((!useSCH) || (nextEvent <= nextChannelSwitch->getArrivalTime())) {
262  if (nextMacEvent->isScheduled()) {

265  scheduleAt(nextEvent,nextMacEvent);
266  DBG_MAC << "Updated nextMacEvent:" << nextMacEvent->getArrivalTime().raw() << std::endl;
267  }
268  else {
269  DBG_MAC << "Too little time in this interval. Will not schedule nextMacEvent" << std::endl;
270  //it is possible that this queue has an txop. we have to revoke it
271  myEDCA[activeChannel]->revokeTxOPs();
272  statsNumTooLittleTime++;
273  }
274  }
275  else {
276  cancelEvent(nextMacEvent);
277  }
278  }
279  if (num == 1 && idleChannel == false && myEDCA[chan]->myQueues[ac].currentBackoff == 0 && chan == activeChannel) {
280  myEDCA[chan]->backoff(ac);

285 void Mac1609_4::handleLowerControl(cMessage* msg) {
286  if (msg->getKind() == MacToPhyInterface::TX_OVER) {
287 
288  DBG_MAC << "Successfully transmitted a packet on " << lastAC << std::endl;
289 
290  phy->setRadioState(Radio::RX);
291 
292  //message was sent
293  //update EDCA queue. go into post-transmit backoff and set cwCur to cwMin
294  myEDCA[activeChannel]->postTransmit(lastAC);
295  //channel just turned idle.
296  //don't set the chan to idle. the PHY layer decides, not us.
297 
298  if (guardActive()) {
299  opp_error("We shouldnt have sent a packet in guard!");
300  }
301  }
302  else if (msg->getKind() == Mac80211pToPhy11pInterface::CHANNEL_BUSY) {
303  channelBusy();
304  }
305  else if (msg->getKind() == Mac80211pToPhy11pInterface::CHANNEL_IDLE) {
306  channelIdle();
307  }
308  else if (msg->getKind() == Decider80211p::BITERROR) {
309  statsSNIRLostPackets++;
310  DBG_MAC << "A packet was not received due to biterrors" << std::endl;
311  }
312  else if (msg->getKind() == Decider80211p::RECWHILESEND) {
313  statsTXRXLostPackets++;
314  DBG_MAC << "A packet was not received because we were sending while receiving" << std::endl;
315  }
316  else if (msg->getKind() == MacToPhyInterface::RADIO_SWITCHING_OVER) {
317  DBG_MAC << "Phylayer said radio switching is done" << std::endl;
318  }
319  else if (msg->getKind() == BaseDecider::PACKET_DROPPED) {
320  phy->setRadioState(Radio::RX);
321  DBG_MAC << "Phylayer said packet was dropped" << std::endl;
322  }
323  else {
324  DBG_MAC << "Invalid control message type (type=NOTHING) : name=" << msg->getName() << " modulesrc=" << msg->getSenderModule()->getFullPath() << "." << std::endl;
325  assert(false);
326  }
327  delete msg;

510 WaveShortMessage* Mac1609_4::EDCA::initiateTransmit(simtime_t lastIdle) {
511 
512  //iterate through the queues to return the packet we want to send
513  WaveShortMessage* pktToSend = NULL;
514 
515  simtime_t idleTime = simTime() - lastIdle;
516 
517  DBG_MAC << "Initiating transmit at " << simTime() << ". I've been idle since " << idleTime << std::endl;
518 
519  for (std::map<t_access_category, EDCAQueue>::iterator iter = myQueues.begin(); iter != myQueues.end(); iter++) {
520  if (iter->second.queue.size() != 0) {
521  if (idleTime >= iter->second.aifsn* SLOTLENGTH_11P + SIFS_11P && iter->second.txOP == true) {
522 
523  DBG_MAC << "Queue " << iter->first << " is ready to send!" << std::endl;
524 
525  iter->second.txOP = false;
526  //this queue is ready to send
527  if (pktToSend == NULL) {
528  pktToSend = iter->second.queue.front();
529  }
530  else {
531  //there was already another packet ready. we have to go increase cw and go into backoff. It's called internal contention and its wonderful
532 
533  statsNumInternalContention++;
534  iter->second.cwCur = std::min(iter->second.cwMax,iter->second.cwCur*2);
535  iter->second.currentBackoff = intuniform(0,iter->second.cwCur);
536  DBG_MAC << "Internal contention for queue " << iter->first << " : "<< iter->second.currentBackoff << ". Increase cwCur to " << iter->second.cwCur << std::endl;
537  }
538  }
539  }
540  }
541 
542  if (pktToSend == NULL) {
543  opp_error("No packet was ready");
544  }
545  return pktToSend;
546 }

548 simtime_t Mac1609_4::EDCA::startContent(simtime_t idleSince,bool guardActive) {
549 
550  DBG_MAC << "Restarting contention." << std::endl;
551 
552  simtime_t nextEvent = -1;
553 
554  simtime_t idleTime = SimTime().setRaw(std::max((int64_t)0,(simTime() - idleSince).raw()));;
555 
556  lastStart = idleSince;
557 
558  DBG_MAC << "Channel is already idle for:" << idleTime << " since " << idleSince << std::endl;
559 
560  //this returns the nearest possible event in this EDCA subsystem after a busy channel
561 
562  for (std::map<t_access_category, EDCAQueue>::iterator iter = myQueues.begin(); iter != myQueues.end(); iter++) {
563  if (iter->second.queue.size() != 0) {
564 
565  /* 1609_4 says that when attempting to send (backoff == 0) when guard is active, a random backoff is invoked */
566 
567  if (guardActive == true && iter->second.currentBackoff == 0) {
568  //cw is not increased
569  iter->second.currentBackoff = intuniform(0,iter->second.cwCur);
570  statsNumBackoff++;
571  }
572 
573  simtime_t DIFS = iter->second.aifsn * SLOTLENGTH_11P + SIFS_11P;
574 
575  //the next possible time to send can be in the past if the channel was idle for a long time, meaning we COULD have sent earlier if we had a packet
576  simtime_t possibleNextEvent = DIFS + iter->second.currentBackoff * SLOTLENGTH_11P;
577 
578 
579  DBG_MAC << "Waiting Time for Queue " << iter->first << ":" << possibleNextEvent << "=" << iter->second.aifsn << " * " << SLOTLENGTH_11P << " + " << SIFS_11P << "+" << iter->second.currentBackoff << "*" << SLOTLENGTH_11P << "; Idle time: " << idleTime << std::endl;
580 
581  if (idleTime > possibleNextEvent) {
582  DBG_MAC << "Could have already send if we had it earlier" << std::endl;
583  //we could have already sent. round up to next boundary
584  simtime_t base = idleSince + DIFS;
585  possibleNextEvent = simTime() - simtime_t().setRaw((simTime() - base).raw() % SLOTLENGTH_11P.raw()) + SLOTLENGTH_11P;
586  }
587  else {
588  //we are gonna send in the future
589  DBG_MAC << "Sending in the future" << std::endl;
590  possibleNextEvent = idleSince + possibleNextEvent;
591  }
592  nextEvent == -1? nextEvent = possibleNextEvent : nextEvent = std::min(nextEvent,possibleNextEvent);
593  }
594  }
595  return nextEvent;
596 }

599  //update all Queues
600 
601  DBG_MAC << "Stopping Contention at " << simTime().raw() << std::endl;
602 
603  simtime_t passedTime = simTime() - lastStart;
604 
605  DBG_MAC << "Channel was idle for " << passedTime << std::endl;
606 
607  lastStart = -1; //indicate that there was no last start
608 
609  for (std::map<t_access_category, EDCAQueue>::iterator iter = myQueues.begin(); iter != myQueues.end(); iter++) {
610  if (iter->second.currentBackoff != 0 || iter->second.queue.size() != 0) {
611  //check how many slots we already waited until the chan became busy
612 
613  int oldBackoff = iter->second.currentBackoff;
614 
615  std::string info;
616  if (passedTime < iter->second.aifsn * SLOTLENGTH_11P + SIFS_11P) {
617  //we didnt even make it one DIFS :(
618  info.append(" No DIFS");
619  }
620  else {
621  //decrease the backoff by one because we made it longer than one DIFS
622  iter->second.currentBackoff--;
623 
624  //check how many slots we waited after the first DIFS
625  int passedSlots = (int)((passedTime - SimTime(iter->second.aifsn * SLOTLENGTH_11P + SIFS_11P)) / SLOTLENGTH_11P);
626 
627  DBG_MAC << "Passed slots after DIFS: " << passedSlots << std::endl;
628 
629 
630  if (iter->second.queue.size() == 0) {
631  //this can be below 0 because of post transmit backoff -> backoff on empty queues will not generate macevents,
632  //we dont want to generate a txOP for empty queues
633  iter->second.currentBackoff -= std::min(iter->second.currentBackoff,passedSlots);
634  info.append(" PostCommit Over");
635  }
636  else {
637  iter->second.currentBackoff -= passedSlots;
638  if (iter->second.currentBackoff <= -1) {
639  if (generateTxOp) {
640  iter->second.txOP = true; info.append(" TXOP");
641  }
642  //else: this packet couldnt be sent because there was too little time. we could have generated a txop, but the channel switched
643  iter->second.currentBackoff = 0;
644  }
645 
646  }
647  }
648  DBG_MAC << "Updating backoff for Queue " << iter->first << ": " << oldBackoff << " -> " << iter->second.currentBackoff << info <<std::endl;
649  }
650  }

729 void Mac1609_4::channelIdle(bool afterSwitch) {
730 
731  DBG_MAC << "Channel turned idle: Switch: " << afterSwitch << std::endl;
732 
733  if (nextMacEvent->isScheduled() == true) {
734  //this rare case can happen when another node's time has such a big offset that the node sent a packet although we already changed the channel
735  //the workaround is not trivial and requires a lot of changes to the phy and decider
736  return;
737  //opp_error("channel turned idle but contention timer was scheduled!");
738  }
739 
740  idleChannel = true;
741 
742  simtime_t delay = 0;
743 
744  //account for 1609.4 guards
745  if (afterSwitch) {
746  // delay = GUARD_INTERVAL_11P;
747  }
748  if (useSCH) {
749  delay += timeLeftTillGuardOver();
750  }
751 
752  //channel turned idle! lets start contention!
753  lastIdle = delay + simTime();
754  statsTotalBusyTime += simTime() - lastBusy;
755 
756  //get next Event from current EDCA subsystem
757  simtime_t nextEvent = myEDCA[activeChannel]->startContent(lastIdle,guardActive());
758  if (nextEvent != -1) {
759  if ((!useSCH) || (nextEvent < nextChannelSwitch->getArrivalTime())) {
760  scheduleAt(nextEvent,nextMacEvent);
761  DBG_MAC << "next Event is at " << nextMacEvent->getArrivalTime().raw() << std::endl;
762  }
763  else {
764  DBG_MAC << "Too little time in this interval. will not schedule macEvent" << std::endl;
765  statsNumTooLittleTime++;
766  myEDCA[activeChannel]->revokeTxOPs();
767  }
768  }
769  else {
770  DBG_MAC << "I don't have any new events in this EDCA sub system" << std::endl;
771  }
