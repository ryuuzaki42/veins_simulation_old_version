252 void TraCIScenarioManager::handleSelfMsg(cMessage *msg) {
253  if (msg == connectAndStartTrigger) {
254  connection = TraCIConnection::connect(host.c_str(), port);
255  commandIfc = new TraCICommandInterface(*connection);
256  init_traci();
257  return;
258  }
259  if (msg == executeOneTimestepTrigger) {
260  if (simTime() > 1) {
261  if (vehicleTypeIds.size()==0) {
262  std::list<std::string> vehTypes = getCommandInterface()->getVehicleTypeIds();
263  for (std::list<std::string>::const_iterator i = vehTypes.begin(); i != vehTypes.end(); ++i) {
264  if (i->compare("DEFAULT_VEHTYPE")!=0) {
265  MYDEBUG << *i << std::endl;
266  vehicleTypeIds.push_back(*i);
267  }
268  }
269  }
270  if (routeIds.size()==0) {
271  std::list<std::string> routes = getCommandInterface()->getRouteIds();
272  for (std::list<std::string>::const_iterator i = routes.begin(); i != routes.end(); ++i) {
273  std::string routeId = *i;
274  if (par("useRouteDistributions").boolValue() == true) {
275  if (std::count(routeId.begin(), routeId.end(), '#') >= 1) {
276  MYDEBUG << "Omitting route " << routeId << " as it seems to be a member of a route distribution (found '#' in name)" << std::endl;
277  continue;
278  }
279  }
280  MYDEBUG << "Adding " << routeId << " to list of possible routes" << std::endl;
281  routeIds.push_back(routeId);
282  }
283  }
284  for (int i = activeVehicleCount + queuedVehicles.size(); i< numVehicles; i++) {
285  insertNewVehicle();
286  }
287  }
288  executeOneTimestep();
289  return;
290  }
291  error("TraCIScenarioManager received unknown self-message");
292 }

297 void TraCIScenarioManager::addModule(std::string nodeId, std::string type, std::string name, std::string displayString, const Coord& position, std::string road_id, double speed, double angle) {
298 
299  if (hosts.find(nodeId) != hosts.end()) error("tried adding duplicate module");
300 
301  if (queuedVehicles.find(nodeId) != queuedVehicles.end()) {
302  queuedVehicles.erase(nodeId);
303  }
304  double option1 = hosts.size() / (hosts.size() + unEquippedHosts.size() + 1.0);
305  double option2 = (hosts.size() + 1) / (hosts.size() + unEquippedHosts.size() + 1.0);
306 
307  if (fabs(option1 - penetrationRate) < fabs(option2 - penetrationRate)) {
308  unEquippedHosts.insert(nodeId);
309  return;
310  }
311 
312  int32_t nodeVectorIndex = nextNodeVectorIndex++;
313 
314  cModule* parentmod = getParentModule();
315  if (!parentmod) error("Parent Module not found");
316 
317  cModuleType* nodeType = cModuleType::get(type.c_str());
318  if (!nodeType) error("Module Type \"%s\" not found", type.c_str());
319 
320  //TODO: this trashes the vectsize member of the cModule, although nobody seems to use it
321  cModule* mod = nodeType->create(name.c_str(), parentmod, nodeVectorIndex, nodeVectorIndex);
322  mod->finalizeParameters();
323  mod->getDisplayString().parse(displayString.c_str());
324  mod->buildInside();
325  mod->scheduleStart(simTime() + updateInterval);
326 
327  // pre-initialize TraCIMobility
328  for (cModule::SubmoduleIterator iter(mod); !iter.end(); iter++) {
329  cModule* submod = iter();
330  ifInetTraCIMobilityCallPreInitialize(submod, nodeId, position, road_id, speed, angle);
331  TraCIMobility* mm = dynamic_cast<TraCIMobility*>(submod);
332  if (!mm) continue;
333  mm->preInitialize(nodeId, position, road_id, speed, angle);
334  }
335 
336  mod->callInitialize();
337  hosts[nodeId] = mod;
338 
339  // post-initialize TraCIMobility
340  for (cModule::SubmoduleIterator iter(mod); !iter.end(); iter++) {
341  cModule* submod = iter();
342  TraCIMobility* mm = dynamic_cast<TraCIMobility*>(submod);
343  if (!mm) continue;
344  mm->changePosition();
345  }

469 double TraCIScenarioManager::traci2omnetAngle(double angle) const {
470 
471  // rotate angle so 0 is east (in TraCI's angle interpretation 0 is south)
472  angle = angle - 90;
473 
474  // convert to rad
475  angle = angle * M_PI / 180.0;
476 
477  // normalize angle to -M_PI <= angle < M_PI
478  while (angle < -M_PI) angle += 2 * M_PI;
479  while (angle >= M_PI) angle -= 2 * M_PI;
480 
481  return angle;
482 }

527 void TraCIScenarioManager::processSimSubscription(std::string objectId, TraCIBuffer& buf) {
528  uint8_t variableNumber_resp; buf >> variableNumber_resp;
529  for (uint8_t j = 0; j < variableNumber_resp; ++j) {
530  uint8_t variable1_resp; buf >> variable1_resp;
531  uint8_t isokay; buf >> isokay;
532  if (isokay != RTYPE_OK) {
533  uint8_t varType; buf >> varType;
534  ASSERT(varType == TYPE_STRING);
535  std::string description; buf >> description;
536  if (isokay == RTYPE_NOTIMPLEMENTED) error("TraCI server reported subscribing to variable 0x%2x not implemented (\"%s\"). Might need newer version.", variable1_resp, description.c_str());
537  error("TraCI server reported error subscribing to variable 0x%2x (\"%s\").", variable1_resp, description.c_str());
538  }
539 
540  if (variable1_resp == VAR_DEPARTED_VEHICLES_IDS) {
541  uint8_t varType; buf >> varType;
542  ASSERT(varType == TYPE_STRINGLIST);
543  uint32_t count; buf >> count;
544  MYDEBUG << "TraCI reports " << count << " departed vehicles." << endl;
545  for (uint32_t i = 0; i < count; ++i) {
546  std::string idstring; buf >> idstring;
547  // adding modules is handled on the fly when entering/leaving the ROI
548  }
549 
550  activeVehicleCount += count;
551  drivingVehicleCount += count;
552 
553  } else if (variable1_resp == VAR_ARRIVED_VEHICLES_IDS) {
554  uint8_t varType; buf >> varType;
555  ASSERT(varType == TYPE_STRINGLIST);
556  uint32_t count; buf >> count;
557  MYDEBUG << "TraCI reports " << count << " arrived vehicles." << endl;
558  for (uint32_t i = 0; i < count; ++i) {

668 void TraCIScenarioManager::processVehicleSubscription(std::string objectId, TraCIBuffer& buf) {
669  bool isSubscribed = (subscribedVehicles.find(objectId) != subscribedVehicles.end());
670  double px;
671  double py;
672  std::string edge;
673  double speed;
674  double angle_traci;
675  int signals;
676  int numRead = 0;
677 
678  uint8_t variableNumber_resp; buf >> variableNumber_resp;
679  for (uint8_t j = 0; j < variableNumber_resp; ++j) {
680  uint8_t variable1_resp; buf >> variable1_resp;
681  uint8_t isokay; buf >> isokay;
682  if (isokay != RTYPE_OK) {
683  uint8_t varType; buf >> varType;
684  ASSERT(varType == TYPE_STRING);
685  std::string errormsg; buf >> errormsg;
686  if (isSubscribed) {
687  if (isokay == RTYPE_NOTIMPLEMENTED) error("TraCI server reported subscribing to vehicle variable 0x%2x not implemented (\"%s\"). Might need newer version.", variable1_resp, errormsg.c_str());
688  error("TraCI server reported error subscribing to vehicle variable 0x%2x (\"%s\").", variable1_resp, errormsg.c_str());
689  }
690  } else if (variable1_resp == ID_LIST) {
691  uint8_t varType; buf >> varType;
692  ASSERT(varType == TYPE_STRINGLIST);
693  uint32_t count; buf >> count;
694  MYDEBUG << "TraCI reports " << count << " active vehicles." << endl;
695  ASSERT(count == drivingVehicleCount);
696  std::set<std::string> drivingVehicles;
697  for (uint32_t i = 0; i < count; ++i) {
698  std::string idstring; buf >> idstring;
699  drivingVehicles.insert(idstring);
700  }
701 
702  // check for vehicles that need subscribing to
703  std::set<std::string> needSubscribe;
704  std::set_difference(drivingVehicles.begin(), drivingVehicles.end(), subscribedVehicles.begin(), subscribedVehicles.end(), std::inserter(needSubscribe, needSubscribe.begin()));
705  for (std::set<std::string>::const_iterator i = needSubscribe.begin(); i != needSubscribe.end(); ++i) {
706  subscribedVehicles.insert(*i);
707  subscribeToVehicleVariables(*i);
708  }
709 
710  // check for vehicles that need unsubscribing from
711  std::set<std::string> needUnsubscribe;
712  std::set_difference(subscribedVehicles.begin(), subscribedVehicles.end(), drivingVehicles.begin(), drivingVehicles.end(), std::inserter(needUnsubscribe, needUnsubscribe.begin()));
713  for (std::set<std::string>::const_iterator i = needUnsubscribe.begin(); i != needUnsubscribe.end(); ++i) {
714  subscribedVehicles.erase(*i);
715  unsubscribeFromVehicleVariables(*i);
716  }
717 
718  } else if (variable1_resp == VAR_POSITION) {
719  uint8_t varType; buf >> varType;
720  ASSERT(varType == POSITION_2D);
721  buf >> px;
722  buf >> py;
723  numRead++;
724  } else if (variable1_resp == VAR_ROAD_ID) {
725  uint8_t varType; buf >> varType;
726  ASSERT(varType == TYPE_STRING);
727  buf >> edge;
728  numRead++;
729  } else if (variable1_resp == VAR_SPEED) {
730  uint8_t varType; buf >> varType;
731  ASSERT(varType == TYPE_DOUBLE);
732  buf >> speed;
733  numRead++;
734  } else if (variable1_resp == VAR_ANGLE) {
735  uint8_t varType; buf >> varType;
736  ASSERT(varType == TYPE_DOUBLE);
737  buf >> angle_traci;
738  numRead++;
739  } else if (variable1_resp == VAR_SIGNALS) {
740  uint8_t varType; buf >> varType;
741  ASSERT(varType == TYPE_INTEGER);
742  buf >> signals;
743  numRead++;
744  } else {
745  error("Received unhandled vehicle subscription result");
746  }
747  }
748 
749  // bail out if we didn't want to receive these subscription results
750  if (!isSubscribed) return;
751 
752  // make sure we got updates for all attributes
753  if (numRead != 5) return;
754 
755  Coord p = traci2omnet(TraCICoord(px, py));
756  if ((p.x < 0) || (p.y < 0)) error("received bad node position (%.2f, %.2f), translated to (%.2f, %.2f)", px, py, p.x, p.y);
757 
758  double angle = traci2omnetAngle(angle_traci);
759 
760  cModule* mod = getManagedModule(objectId);
761 
762  // is it in the ROI?
763  bool inRoi = isInRegionOfInterest(TraCICoord(px, py), edge, speed, angle);
764  if (!inRoi) {
765  if (mod) {
766  deleteModule(objectId);
767  MYDEBUG << "Vehicle #" << objectId << " left region of interest" << endl;
768  }
769  else if(unEquippedHosts.find(objectId) != unEquippedHosts.end()) {
770  unEquippedHosts.erase(objectId);
771  MYDEBUG << "Vehicle (unequipped) # " << objectId<< " left region of interest" << endl;
772  }
773  return;
774  }
775 
776  if (isModuleUnequipped(objectId)) {
777  return;
778  }
779 
780  if (!mod) {
781  // no such module - need to create
782  addModule(objectId, moduleType, moduleName, moduleDisplayString, p, edge, speed, angle);
783  MYDEBUG << "Added vehicle #" << objectId << endl;
784  } else {
785  // module existed - update position
786  for (cModule::SubmoduleIterator iter(mod); !iter.end(); iter++) {
787  cModule* submod = iter();
788  ifInetTraCIMobilityCallNextPosition(submod, p, edge, speed, angle);
789  TraCIMobility* mm = dynamic_cast<TraCIMobility*>(submod);
790  if (!mm) continue;
791  MYDEBUG << "module " << objectId << " moving to " << p.x << "," << p.y << endl;
792  mm->nextPosition(p, edge, speed, angle);
793  }
794  }
795 
796 }
