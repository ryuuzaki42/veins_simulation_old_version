370 void PhyLayer80211p::handleSelfMessage(cMessage* msg) {
371 
372  switch(msg->getKind()) {
373  //transmission overBasePhyLayer::
374  case TX_OVER: {
375  assert(msg == txOverTimer);
376  sendControlMsgToMac(new cMessage("Transmission over", TX_OVER));
377  //check if there is another packet on the chan, and change the chan-state to idle
378  Decider80211p* dec = dynamic_cast<Decider80211p*>(decider);
379  assert(dec);
380  if (dec->cca(simTime(),NULL)) {
381  //chan is idle
382  DBG << "Channel idle after transmit!\n";
383  dec->setChannelIdleStatus(true);
384 
385  }
386  else {
387  DBG << "Channel not yet idle after transmit!\n";
388  }
389  break;
390  }
391  //radio switch over
392  case RADIO_SWITCHING_OVER:
393  assert(msg == radioSwitchingOverTimer);
394  BasePhyLayer::finishRadioSwitching();
395  break;
396 
397  //AirFrame
398  case AIR_FRAME:
399  BasePhyLayer::handleAirFrame(static_cast<AirFrame*>(msg));
400  break;
401 
402  //ChannelSenseRequest
403  case CHANNEL_SENSE_REQUEST:
404  BasePhyLayer::handleChannelSenseRequest(msg);
405  break;
406 
407  default:
408  break;
409  }
410 }
411 AirFrame *PhyLayer80211p::encapsMsg(cPacket *macPkt)
412 {
413  // the cMessage passed must be a MacPacket... but no cast needed here
414  // MacPkt* pkt = static_cast<MacPkt*>(msg);
415 
416  // ...and must always have a ControlInfo attached (contains Signal)
417  cObject* ctrlInfo = macPkt->removeControlInfo();
418  assert(ctrlInfo);
419 
420  // create the new AirFrame
421  AirFrame* frame = new AirFrame11p(macPkt->getName(), AIR_FRAME);
422 
423  // Retrieve the pointer to the Signal-instance from the ControlInfo-instance.
424  // We are now the new owner of this instance.
425  Signal* s = MacToPhyControlInfo::getSignalFromControlInfo(ctrlInfo);
426  // make sure we really obtained a pointer to an instance
427  assert(s);
428 
429  // set the members
430  assert(s->getDuration() > 0);
431  frame->setDuration(s->getDuration());
432  // copy the signal into the AirFrame
433  frame->setSignal(*s);
434  //set priority of AirFrames above the normal priority to ensure
435  //channel consistency (before any thing else happens at a time
436  //point t make sure that the channel has removed every AirFrame
437  //ended at t and added every AirFrame started at t)
438  frame->setSchedulingPriority(airFramePriority);
439  frame->setProtocolId(myProtocolId());
440  frame->setBitLength(headerLength);
441  frame->setId(world->getUniqueAirFrameId());
442  frame->setChannel(radio->getCurrentChannel());
443 
444 
445  // pointer and Signal not needed anymore
446  delete s;
447  s = 0;
448 
449  // delete the Control info
450  delete ctrlInfo;
451  ctrlInfo = 0;
452 
453  frame->encapsulate(macPkt);
454 
455  // --- from here on, the AirFrame is the owner of the MacPacket ---
456  macPkt = 0;
457  coreEV <<"AirFrame encapsulated, length: " << frame->getBitLength() << "\n";
458 
459  return frame;
460 }
