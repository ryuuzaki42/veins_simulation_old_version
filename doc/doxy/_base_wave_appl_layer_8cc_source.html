314 //WaveShortMessage* BaseWaveApplLayer::prepareWSM(std::string name, int lengthBits, t_channel channel, int priority, int rcvId, int serial) {
315 WaveShortMessage* BaseWaveApplLayer::prepareWSM(std::string name, int lengthBits, t_channel channel, int priority, unsigned int rcvId, int serial) {
316  WaveShortMessage* wsm = new WaveShortMessage(name.c_str());
317  wsm->addBitLength(headerLength);
318  wsm->addBitLength(lengthBits);
319 
320  switch (channel) {
321  case type_SCH: wsm->setChannelNumber(Channels::SCH1); break; //will be rewritten at Mac1609_4 to actual Service Channel. This is just so no controlInfo is needed
322  case type_CCH: wsm->setChannelNumber(Channels::CCH); break;
323  }
324  wsm->setPsid(0);
325  wsm->setPriority(priority);
326  wsm->setWsmVersion(1);
327  wsm->setTimestamp(simTime());
328  wsm->setSenderAddress(myId);
329  wsm->setRecipientAddress(rcvId);
330  wsm->setSenderPos(curPosition);
331  wsm->setSerial(serial);
332 
333  if (name == "beacon_minicurso") {
334  // Change Minicurso_UFPI
335  wsm->setRoadId(TraCIMobilityAccess().get(getParentModule()) ->getRoadId().c_str());
336  wsm->setSenderSpeed(TraCIMobilityAccess(). get(getParentModule())->getSpeed());
337  //DBG << "\n ttt" << wsm->getSenderSpeed() << " ttt\n";
338  DBG << "Creating Beacon with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
339  }else if (name == "beacon") {
340  DBG << "Creating Beacon with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
341  }else if (name == "data") {
342  DBG << "Creating Data with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
343  }
344  return wsm;
345 }

415 void BaseWaveApplLayer::handleSelfMsg(cMessage* msg) {
416  switch (msg->getKind()) {
417  case SEND_BEACON_EVT: {
418  sendWSM(prepareWSM("beacon", beaconLengthBits, type_CCH, beaconPriority, 0, -1));
419  scheduleAt(simTime() + par("beaconInterval").doubleValue(), sendBeaconEvt);
420  break;
421  }
422  case SEND_BEACON_EVT_minicurso: {
423  sendWSM(prepareWSM("beacon_minicurso", beaconLengthBits, type_CCH, beaconPriority, 0, -1));
424  scheduleAt(simTime() + par("beaconInterval").doubleValue(), sendBeaconEvt);
425  break;
426  }
427  case SEND_BEACON_EVT_epidemic: {
428  //prepareWSM(std::string name, int lengthBits, t_channel channel, int priority, int rcvId, int serial)
429  //I our implementation, if rcvId = BROADCAST then we are broadcasting beacons. Otherwise, this parameter must be instantiated with the receiver address
430  sendWSM(prepareWSM_epidemic("beacon", beaconLengthBits, type_CCH, beaconPriority, BROADCAST, -1));
431  scheduleAt(simTime() + par("beaconInterval").doubleValue(), sendBeaconEvt);
432  break;
433  }
434  default: {
435  if (msg)
436  DBG << "APP: Error: Got Self Message of unknown kind! Name: " << msg->getName() << endl;
437  break;
438  }
439  }
