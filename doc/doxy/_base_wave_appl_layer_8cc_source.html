292 //WaveShortMessage* BaseWaveApplLayer::prepareWSM(std::string name, int lengthBits, t_channel channel, int priority, int rcvId, int serial) {
293 WaveShortMessage* BaseWaveApplLayer::prepareWSM(std::string name, int lengthBits, t_channel channel, int priority, unsigned int rcvId, int serial) {
294  WaveShortMessage* wsm = new WaveShortMessage(name.c_str());
295  wsm->addBitLength(headerLength);
296  wsm->addBitLength(lengthBits);
297 
298  switch (channel) {
299  case type_SCH: wsm->setChannelNumber(Channels::SCH1); break; //will be rewritten at Mac1609_4 to actual Service Channel. This is just so no controlInfo is needed
300  case type_CCH: wsm->setChannelNumber(Channels::CCH); break;
301  }
302  wsm->setPsid(0);
303  wsm->setPriority(priority);
304  wsm->setWsmVersion(1);
305  wsm->setTimestamp(simTime());
306  wsm->setSenderAddress(myId);
307  wsm->setRecipientAddress(rcvId);
308  wsm->setSenderPos(curPosition);
309  wsm->setSerial(serial);
310 
311  if (name == "beacon_minicurso") {
312  // Change Minicurso_UFPI
313  wsm->setRoadId(TraCIMobilityAccess().get(getParentModule()) ->getRoadId().c_str());
314  wsm->setSenderSpeed(TraCIMobilityAccess(). get(getParentModule())->getSpeed());
315  //DBG << "\n ttt" << wsm->getSenderSpeed() << " ttt\n";
316  DBG << "Creating Beacon with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
317  }else if (name == "beacon") {
318  DBG << "Creating Beacon with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
319  }else if (name == "data") {
320  DBG << "Creating Data with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
321  }
322  return wsm;
323 }
