236 WaveShortMessage* BaseWaveApplLayer::prepareWSM(std::string name, int lengthBits, t_channel channel, int priority, unsigned int rcvId, int serial) {
237  WaveShortMessage* wsm = new WaveShortMessage(name.c_str());
238  wsm->addBitLength(headerLength);
239  wsm->addBitLength(lengthBits);
240 
241  switch (channel) {
242  case type_SCH: wsm->setChannelNumber(Channels::SCH1); break; //will be rewritten at Mac1609_4 to actual Service Channel. This is just so no controlInfo is needed
243  case type_CCH: wsm->setChannelNumber(Channels::CCH); break;
244  }
245  wsm->setPsid(0);
246  wsm->setPriority(priority);
247  wsm->setWsmVersion(1);
248  wsm->setTimestamp(simTime());
249  wsm->setSenderAddress(myId);
250  wsm->setRecipientAddress(rcvId);
251  wsm->setSenderPos(curPosition);
252  wsm->setSerial(serial);
253 
254  if (name == "beacon_minicurso") {
255  // Change Minicurso_UFPI
256  wsm->setRoadId(TraCIMobilityAccess().get(getParentModule()) ->getRoadId().c_str());
257  wsm->setSenderSpeed(TraCIMobilityAccess(). get(getParentModule())->getSpeed());
258  //DBG << "\n ttt" << wsm->getSenderSpeed() << " ttt\n";
259  DBG << "Creating Beacon with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
260  }else if (name == "beacon") {
261  DBG << "Creating Beacon with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
262  }else if (name == "data") {
263  DBG << "Creating Data with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
264  }
265  return wsm;
266 }
