75 void BaseLayer::handleMessage(cMessage* msg)
76 {
77  if (msg->isSelfMessage()){
78  handleSelfMsg(msg);
79  } else if(msg->getArrivalGateId()==upperLayerIn) {
80  recordPacket(PassedMessage::INCOMING,PassedMessage::UPPER_DATA,msg);
81  handleUpperMsg(msg);
82  } else if(msg->getArrivalGateId()==upperControlIn) {
83  recordPacket(PassedMessage::INCOMING,PassedMessage::UPPER_CONTROL,msg);
84  handleUpperControl(msg);
85  } else if(msg->getArrivalGateId()==lowerControlIn){
86  recordPacket(PassedMessage::INCOMING,PassedMessage::LOWER_CONTROL,msg);
87  handleLowerControl(msg);
88  } else if(msg->getArrivalGateId()==lowerLayerIn) {
89  recordPacket(PassedMessage::INCOMING,PassedMessage::LOWER_DATA,msg);
90  handleLowerMsg(msg);
91  }
92  else if(msg->getArrivalGateId()==-1) {
93  /* Classes extending this class may not use all the gates, f.e.
94  * BaseApplLayer has no upper gates. In this case all upper gate-
95  * handles are initialized to -1. When getArrivalGateId() equals -1,
96  * it would be wrong to forward the message to one of these gates,
97  * as they actually don't exist, so raise an error instead.
98  */
99  opp_error("No self message and no gateID?? Check configuration.");
100  } else {
101  /* msg->getArrivalGateId() should be valid, but it isn't recognized
102  * here. This could signal the case that this class is extended
103  * with extra gates, but handleMessage() isn't overridden to
104  * check for the new gate(s).
105  */
106  opp_error("Unknown gateID?? Check configuration or override handleMessage().");
107  }
108 }
