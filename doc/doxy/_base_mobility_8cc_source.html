454 BaseMobility::BorderHandling BaseMobility::checkIfOutside( Coord targetPos,
455  Coord& borderStep )
456 {
457  BorderHandling outside = NOWHERE;
458 
459  // Testing x-value
460  if (targetPos.x < 0){
461  borderStep.x = (-move.getStartPos().x);
462  outside = X_SMALLER;
463  }
464  else if (targetPos.x >= playgroundSizeX()){
465  borderStep.x = (playgroundSizeX() - move.getStartPos().x);
466  outside = X_BIGGER;
467  }
468 
469  // Testing y-value
470  if (targetPos.y < 0){
471  borderStep.y = (-move.getStartPos().y);
472 
473  if( outside == NOWHERE
474  || fabs(borderStep.x/move.getDirection().x)
475  > fabs(borderStep.y/move.getDirection().y) )
476  {
477  outside = Y_SMALLER;
478  }
479  }
480  else if (targetPos.y >= playgroundSizeY()){
481  borderStep.y = (playgroundSizeY() - move.getStartPos().y);
482 
483  if( outside == NOWHERE
484  || fabs(borderStep.x/move.getDirection().x)
485  > fabs(borderStep.y/move.getDirection().y) )
486  {
487  outside = Y_BIGGER;
488  }
489  }
490 
491  // Testing z-value
492  if (!world->use2D())
493  {
494  // going to reach the lower z-border
495  if (targetPos.z < 0)
496  {
497  borderStep.z = (-move.getStartPos().z);
498 
499  // no border reached so far
500  if( outside==NOWHERE )
501  {
502  outside = Z_SMALLER;
503  }
504  // an y-border is reached earliest so far, test whether z-border
505  // is reached even earlier
506  else if( (outside == Y_SMALLER || outside == Y_BIGGER)
507  && fabs(borderStep.y/move.getDirection().y)
508  > fabs(borderStep.z/move.getDirection().z) )
509  {
510  outside = Z_SMALLER;
511  }
512  // an x-border is reached earliest so far, test whether z-border
513  // is reached even earlier
514  else if( (outside == X_SMALLER || outside == X_BIGGER)
515  && fabs(borderStep.x/move.getDirection().x)
516  > fabs(borderStep.z/move.getDirection().z) )
517  {
518  outside = Z_SMALLER;
519  }
520 
521  }
522  // going to reach the upper z-border
523  else if (targetPos.z >= playgroundSizeZ())
524  {
525  borderStep.z = (playgroundSizeZ() - move.getStartPos().z);
526 
527  // no border reached so far
528  if( outside==NOWHERE )
529  {
530  outside = Z_BIGGER;
531  }
532  // an y-border is reached earliest so far, test whether z-border
533  // is reached even earlier
534  else if( (outside==Y_SMALLER || outside==Y_BIGGER)
535  && fabs(borderStep.y/move.getDirection().y)
536  > fabs(borderStep.z/move.getDirection().z) )
537  {
538  outside = Z_BIGGER;
539  }
540  // an x-border is reached earliest so far, test whether z-border
541  // is reached even earlier
542  else if( (outside==X_SMALLER || outside==X_BIGGER)
543  && fabs(borderStep.x/move.getDirection().x)
544  > fabs(borderStep.z/move.getDirection().z) )
545  {
546  outside = Z_BIGGER;
547  }
548 
549 
550  }
551  }
552 
553  coreEV << "checkIfOutside, outside="<<outside<<" borderStep: " << borderStep.info() << endl;
554 
555  return outside;
556 }

564

573  borderStep.y = (factor * move.getDirection().y);

575  {
576  borderStep.z = (factor * move.getDirection().z); // 3D case
577  }
578 
579  if( policy == WRAP )
580  {
581  borderStart.x = (playgroundSizeX());
582  borderStart.y = (move.getStartPos().y + borderStep.y);
583  if (!world->use2D())
584  {
585  borderStart.z = (move.getStartPos().z
586  + borderStep.z); // 3D case
587  }
588  }
589  break;
590 
591  case X_BIGGER:
592  factor = borderStep.x / move.getDirection().x;
593  borderStep.y = (factor * move.getDirection().y);
594  if (!world->use2D())
595  {
596  borderStep.z = (factor * move.getDirection().z); // 3D case
597  }
598 
599  if( policy == WRAP )
600  {
601  borderStart.x = (0);
602  borderStart.y = (move.getStartPos().y + borderStep.y);
603  if (!world->use2D())
604  {
605  borderStart.z = (move.getStartPos().z
606  + borderStep.z); // 3D case
607  }
608  }
609  break;
610 
611  case Y_SMALLER:
612  factor = borderStep.y / move.getDirection().y;
613  borderStep.x = (factor * move.getDirection().x);
614  if (!world->use2D())
615  {
616  borderStep.z = (factor * move.getDirection().z); // 3D case
617  }
618 
619  if( policy == WRAP )
620  {
621  borderStart.y = (playgroundSizeY());
622  borderStart.x = (move.getStartPos().x + borderStep.x);
623  if (!world->use2D())
624  {
625  borderStart.z = (move.getStartPos().z
626  + borderStep.z); // 3D case
627  }
628  }
629  break;
630 
631  case Y_BIGGER:
632  factor = borderStep.y / move.getDirection().y;
633  borderStep.x = (factor * move.getDirection().x);
634  if (!world->use2D())
635  {
636  borderStep.z = (factor * move.getDirection().z); // 3D case
637  }
638 
639  if( policy == WRAP )
640  {
641  borderStart.y = (0);
642  borderStart.x = (move.getStartPos().x + borderStep.x);
643  if (!world->use2D())
644  {
645  borderStart.z = (move.getStartPos().z
646  + borderStep.z); // 3D case
647  }
648  }
649  break;
650 
651  case Z_SMALLER: // here we are definitely in 3D
652  factor = borderStep.z / move.getDirection().z;
653  borderStep.x = (factor * move.getDirection().x);
654  borderStep.y = (factor * move.getDirection().y);
655 
656  if( policy == WRAP )
657  {
658  borderStart.z = (playgroundSizeZ());
659  borderStart.x = (move.getStartPos().x + borderStep.x);
660  borderStart.y = (move.getStartPos().y + borderStep.y);
661  }
662  break;
663 
664  case Z_BIGGER: // here we are definitely in 3D
665  factor = borderStep.z / move.getDirection().z;
666  borderStep.x = (factor * move.getDirection().x);
667  borderStep.y = (factor * move.getDirection().y);
668 
669  if( policy == WRAP )
670  {
671  borderStart.z = (0);
672  borderStart.x = (move.getStartPos().x + borderStep.x);
673  borderStart.y = (move.getStartPos().y + borderStep.y);
674  }
675  break;
676 
677  default:
678  factor = 0;
679  error("invalid state in goToBorder switch!");
680  break;
681  }
682 
683  coreEV << "goToBorder: startPos: " << move.getStartPos().info()
684  << " borderStep: " << borderStep.info()
685  << " borderStart: " << borderStart.info()
686  << " factor: " << factor << endl;

691 bool BaseMobility::handleIfOutside(BorderPolicy policy, Coord& stepTarget,
692  Coord& targetPos, Coord& step,
693  double& angle)
694 {
695  // where did the host leave the playground?
696  BorderHandling wo;
697 
698  // step to reach the border
699  Coord borderStep;
700 
701  wo = checkIfOutside(stepTarget, borderStep);
702 
703  // just return if the next step is not outside the playground
704  if( wo == NOWHERE )
705  return false;
706 
707  coreEV << "handleIfOutside:stepTarget = " << stepTarget.info() << endl;
708 
709  // new start position after the host reaches the border
710  Coord borderStart;
711  // new direction the host has to move to
712  Coord borderDirection;
713  // time to reach the border
714  simtime_t borderInterval;
715 
716  coreEV << "old values: stepTarget: " << stepTarget.info()
717  << " step: " << step.info()
718  << " targetPos: " << targetPos.info()
719  << " angle: " << angle << endl;
720 
721  // which border policy is to be followed
722  switch (policy){
723  case REFLECT:
724  reflectIfOutside( wo, stepTarget, targetPos, step, angle );
725  break;
726  case WRAP:
727  wrapIfOutside( wo, stepTarget, targetPos );
728  break;
729  case PLACERANDOMLY:
730  placeRandomlyIfOutside( targetPos );
731  break;
732  case RAISEERROR:
733  break;
734  }
735 
736  coreEV << "new values: stepTarget: " << stepTarget.info()
737  << " step: " << step.info()
738  << " angle: " << angle << endl;
739 
740  // calculate the step to reach the border
741  goToBorder(policy, wo, borderStep, borderStart);
742 
743  // calculate the time to reach the border
744  borderInterval = (borderStep.length()) / move.getSpeed();
745 
746  // calculate new start position
747  // NOTE: for WRAP this is done in goToBorder
748  switch( policy ){
749  case REFLECT:
750  {
751  borderStart = move.getStartPos() + borderStep;
752  double d = stepTarget.distance( borderStart );
753  borderDirection = (stepTarget - borderStart) / d;
754  break;
