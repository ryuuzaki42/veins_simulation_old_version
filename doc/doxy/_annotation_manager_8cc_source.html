116 void AnnotationManager::addFromXml(cXMLElement* xml) {
117  std::string rootTag = xml->getTagName();
118  ASSERT (rootTag == "annotations");
119 
120  cXMLElementList list = xml->getChildren();
121  for (cXMLElementList::const_iterator i = list.begin(); i != list.end(); ++i) {
122  cXMLElement* e = *i;
123 
124  std::string tag = e->getTagName();
125 
126  if (tag == "point") {
127  ASSERT(e->getAttribute("text"));
128  std::string text = e->getAttribute("text");
129  ASSERT(e->getAttribute("color"));
130  std::string color = e->getAttribute("color");
131  ASSERT(e->getAttribute("shape"));
132  std::string shape = e->getAttribute("shape");
133  std::vector<std::string> points = cStringTokenizer(shape.c_str(), " ").asVector();
134  ASSERT(points.size() == 2);
135 
136  std::vector<double> p1a = cStringTokenizer(points[0].c_str(), ",").asDoubleVector();
137  ASSERT(p1a.size() == 2);
138 
139  drawPoint(Coord(p1a[0], p1a[1]), color, text);
140  }
141  else if (tag == "line") {
142  ASSERT(e->getAttribute("color"));
143  std::string color = e->getAttribute("color");
144  ASSERT(e->getAttribute("shape"));
145  std::string shape = e->getAttribute("shape");
146  std::vector<std::string> points = cStringTokenizer(shape.c_str(), " ").asVector();
147  ASSERT(points.size() == 2);
148 
149  std::vector<double> p1a = cStringTokenizer(points[0].c_str(), ",").asDoubleVector();
150  ASSERT(p1a.size() == 2);
151  std::vector<double> p2a = cStringTokenizer(points[1].c_str(), ",").asDoubleVector();
152  ASSERT(p2a.size() == 2);
153 
154  drawLine(Coord(p1a[0], p1a[1]), Coord(p2a[0], p2a[1]), color);
155  }
156  else if (tag == "poly") {
157  ASSERT(e->getAttribute("color"));
158  std::string color = e->getAttribute("color");
159  ASSERT(e->getAttribute("shape"));
160  std::string shape = e->getAttribute("shape");
161  std::vector<std::string> points = cStringTokenizer(shape.c_str(), " ").asVector();
162  ASSERT(points.size() >= 2);
163 
164  std::vector<Coord> coords;
165  for (std::vector<std::string>::const_iterator i = points.begin(); i != points.end(); ++i) {
166  std::vector<double> pa = cStringTokenizer(i->c_str(), ",").asDoubleVector();
167  ASSERT(pa.size() == 2);
168  coords.push_back(Coord(pa[0], pa[1]));
169  }
170 
171  drawPolygon(coords, color);
172  }
173  else {
174  error("while reading annotations xml: expected 'line' or 'poly', but got '%s'", tag.c_str());
175  }
176  }
177 
178 }

287 cModule* AnnotationManager::createDummyModuleLine(Coord p1, Coord p2, std::string color) {
288  double w = std::abs(p2.x - p1.x);
289  double h = std::abs(p2.y - p1.y);
290  double px = 0;
291  if (p1.x <= p2.x) {
292  px = p1.x + 0.5 * w;
293  } else {
294  px = p2.x + 0.5 * w;
295  w = -w;
296  }
297  double py = 0;
298  if (p1.y <= p2.y) {
299  py = p1.y + 0.5 * h;
300  } else {
301  py = p2.y + 0.5 * h;
302  h = -h;
303  }
304 
305  std::stringstream ss;
306  ss << "p=" << px << "," << py << ";b=" << w << ", " << h << ",polygon," << color << "," << color << ",1";
307  std::string displayString = ss.str();
308 
309  return createDummyModule(displayString);

313 void AnnotationManager::show(const Annotation* annotation) {
314 
315  if (annotation->dummyObjects.size() > 0) return;
316 
317  if (const Point* o = dynamic_cast<const Point*>(annotation)) {
318 
319  if (ev.isGUI()) {
320  // no corresponding TkEnv representation
321  }
322 
323  TraCIScenarioManager* traci = TraCIScenarioManagerAccess().get();
324  if (traci && traci->isConnected()) {
325  std::stringstream nameBuilder; nameBuilder << o->text << " " << ev.getUniqueNumber();
326  traci->getCommandInterface()->addPoi(nameBuilder.str(), "Annotation", TraCIColor::fromTkColor(o->color), 6, traci->omnet2traci(o->pos));
327  annotation->traciPoiIds.push_back(nameBuilder.str());
328  }
329  }
330  else if (const Line* l = dynamic_cast<const Line*>(annotation)) {
331 
332  if (ev.isGUI()) {
333  cModule* mod = createDummyModuleLine(l->p1, l->p2, l->color);
334  annotation->dummyObjects.push_back(mod);
335  }
336 
337  TraCIScenarioManager* traci = TraCIScenarioManagerAccess().get();
338  if (traci && traci->isConnected()) {
339  std::list<Coord> coords; coords.push_back(l->p1); coords.push_back(l->p2);
340  std::stringstream nameBuilder; nameBuilder << "Annotation" << ev.getUniqueNumber();
341  traci->getCommandInterface()->addPolygon(nameBuilder.str(), "Annotation", TraCIColor::fromTkColor(l->color), false, 5, traci->omnet2traci(coords));
342  annotation->traciLineIds.push_back(nameBuilder.str());
343  }
344  }
345  else if (const Polygon* p = dynamic_cast<const Polygon*>(annotation)) {
346 
347  ASSERT(p->coords.size() >= 2);
348 
349  if (ev.isGUI()) {
350  Coord lastCoords = *p->coords.rbegin();
351  for (std::list<Coord>::const_iterator i = p->coords.begin(); i != p->coords.end(); ++i) {
352  Coord c1 = *i;
353  Coord c2 = lastCoords;
354  lastCoords = c1;
355 
356  cModule* mod = createDummyModuleLine(c1, c2, p->color);
357  annotation->dummyObjects.push_back(mod);
358  }
359  }
360 
361  TraCIScenarioManager* traci = TraCIScenarioManagerAccess().get();
362  if (traci && traci->isConnected()) {
363  std::stringstream nameBuilder; nameBuilder << "Annotation" << ev.getUniqueNumber();
364  traci->getCommandInterface()->addPolygon(nameBuilder.str(), "Annotation", TraCIColor::fromTkColor(p->color), false, 4, traci->omnet2traci(p->coords));
365  annotation->traciPolygonsIds.push_back(nameBuilder.str());
366  }
367 
368  }
369  else {
370  error("unknown Annotation type");
371  }
372 
373 }
