71 void ObstacleControl::addFromXml(cXMLElement* xml) {
72  std::string rootTag = xml->getTagName();
73  ASSERT (rootTag == "obstacles");
74 
75  cXMLElementList list = xml->getChildren();
76  for (cXMLElementList::const_iterator i = list.begin(); i != list.end(); ++i) {
77  cXMLElement* e = *i;
78 
79  std::string tag = e->getTagName();
80  ASSERT(tag == "poly");
81 
82  // <poly id="building#0" type="building" color="#F00" shape="16,0 8,13.8564 -8,13.8564 -16,0 -8,-13.8564 8,-13.8564" />
83  ASSERT(e->getAttribute("id"));
84  std::string id = e->getAttribute("id");
85  ASSERT(e->getAttribute("type"));
86  std::string type = e->getAttribute("type");
87  ASSERT(e->getAttribute("color"));
88  std::string color = e->getAttribute("color");
89  ASSERT(e->getAttribute("shape"));
90  std::string shape = e->getAttribute("shape");
91 
92  double attenuationPerWall = 50;
93  double attenuationPerMeter = 1;
94  if (type == "building") { attenuationPerWall = 50; attenuationPerMeter = 1; }
95  else error("unknown obstacle type: %s", type.c_str());
96  Obstacle obs(id, attenuationPerWall, attenuationPerMeter);
97  std::vector<Coord> sh;
98  cStringTokenizer st(shape.c_str());
99  while (st.hasMoreTokens()) {
100  std::string xy = st.nextToken();
101  std::vector<double> xya = cStringTokenizer(xy.c_str(), ",").asDoubleVector();
102  ASSERT(xya.size() == 2);
103  sh.push_back(Coord(xya[0], xya[1]));
104  }
105  obs.setShape(sh);
106  add(obs);
107 
108  }
109

113  Obstacle* o = new Obstacle(obstacle);

115  size_t fromRow = std::max(0, int(o->getBboxP1().x / GRIDCELL_SIZE));
116  size_t toRow = std::max(0, int(o->getBboxP2().x / GRIDCELL_SIZE));
117  size_t fromCol = std::max(0, int(o->getBboxP1().y / GRIDCELL_SIZE));
118  size_t toCol = std::max(0, int(o->getBboxP2().y / GRIDCELL_SIZE));
119  for (size_t row = fromRow; row <= toRow; ++row) {
120  for (size_t col = fromCol; col <= toCol; ++col) {
121  if (obstacles.size() < col+1) obstacles.resize(col+1);
122  if (obstacles[col].size() < row+1) obstacles[col].resize(row+1);
123  (obstacles[col])[row].push_back(o);
124  }
125  }
126 
127  // visualize using AnnotationManager
128  if (annotations) o->visualRepresentation = annotations->drawPolygon(o->getShape(), "red", annotationGroup);

153 double ObstacleControl::calculateAttenuation(const Coord& senderPos, const Coord& receiverPos) const {
154  Enter_Method_Silent();
155 
156  // return cached result, if available
157  CacheKey cacheKey(senderPos, receiverPos);
158  CacheEntries::const_iterator cacheEntryIter = cacheEntries.find(cacheKey);
159  if (cacheEntryIter != cacheEntries.end()) return cacheEntryIter->second;
160 
161  // calculate bounding box of transmission
162  Coord bboxP1 = Coord(std::min(senderPos.x, receiverPos.x), std::min(senderPos.y, receiverPos.y));
163  Coord bboxP2 = Coord(std::max(senderPos.x, receiverPos.x), std::max(senderPos.y, receiverPos.y));
164 
165  size_t fromRow = std::max(0, int(bboxP1.x / GRIDCELL_SIZE));
166  size_t toRow = std::max(0, int(bboxP2.x / GRIDCELL_SIZE));
167  size_t fromCol = std::max(0, int(bboxP1.y / GRIDCELL_SIZE));
168  size_t toCol = std::max(0, int(bboxP2.y / GRIDCELL_SIZE));
169 
170  std::set<Obstacle*> processedObstacles;
171  double factor = 1;
172  for (size_t col = fromCol; col <= toCol; ++col) {
173  if (col >= obstacles.size()) break;
174  for (size_t row = fromRow; row <= toRow; ++row) {
175  if (row >= obstacles[col].size()) break;
176  const ObstacleGridCell& cell = (obstacles[col])[row];
177  for (ObstacleGridCell::const_iterator k = cell.begin(); k != cell.end(); ++k) {
178 
179  Obstacle* o = *k;
180 
181  if (processedObstacles.find(o) != processedObstacles.end()) continue;
182  processedObstacles.insert(o);
183 
184  // bail if bounding boxes cannot overlap
185  if (o->getBboxP2().x < bboxP1.x) continue;
186  if (o->getBboxP1().x > bboxP2.x) continue;
187  if (o->getBboxP2().y < bboxP1.y) continue;
188  if (o->getBboxP1().y > bboxP2.y) continue;
189 
190  double factorOld = factor;
191 
192  factor *= o->calculateAttenuation(senderPos, receiverPos);
193 
194  // draw a "hit!" bubble
195  if (annotations && (factor != factorOld)) annotations->drawBubble(o->getBboxP1(), "hit");
196 
197  // bail if attenuation is already extremely high
198  if (factor < 1e-30) break;
199 
200  }
201  }
202  }
203 
204  // cache result
205  if (cacheEntries.size() >= 1000) cacheEntries.clear();
206  cacheEntries[cacheKey] = factor;
207 
208  return factor;
209 }
