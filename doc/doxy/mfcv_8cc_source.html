116  break;
117  }
118  default: {
119  if (msg)
120  DBG << "APP: Error: Got Self Message of unknown kind! Name: " << msg->getName() << endl;
121  break;
122  }
123  }

148 WaveShortMessage* mfcv::prepareWSM_node(std::string name, int lengthBits, t_channel channel, int priority, int rcvId, int serial) {
149  WaveShortMessage* wsm = new WaveShortMessage(name.c_str());
150  wsm->addBitLength(headerLength);
151  wsm->addBitLength(lengthBits);
152 
153  switch (channel) {
154  case type_SCH: wsm->setChannelNumber(Channels::SCH1); break; //will be rewritten at Mac1609_4 to actual Service Channel. This is just so no controlInfo is needed
155  case type_CCH: wsm->setChannelNumber(Channels::CCH); break;
156  }
157  wsm->setPsid(0);
158  wsm->setPriority(priority);
159  wsm->setWsmVersion(1);
160 
161  //
162  wsm->setRoadId(traci->getRoadId().c_str());
163  wsm->setSenderSpeed(traci->getSpeed());
164  wsm->setVehicleId(traci->getId());
165 
166  // ver como definir o id, traci->getId()
167  int veh_id = traci->getId();
168  if (veh_id = 10) {
169  wsm->setCategory(1);
170  }
171  else if (veh_id = 16) {
172  wsm->setCategory(2);
173  }
174  else {
175  wsm->setCategory(10);
176  }
177 
178 
179  wsm->setTimestamp(simTime());
180  wsm->setSenderAddress(myId);
181  wsm->setRecipientAddress(rcvId);
182  wsm->setSenderPos(curPosition);
183  wsm->setSerial(serial);
184 
185  if (name == "beacon") {
186  DBG << "Creating Beacon with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
187  }
188  else if (name == "beacon_node") {
189  DBG << "Creating Beacon_node with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
190  }
191  else if (name == "data") {
192  DBG << "Creating Data with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
193  }
194  return wsm;
195 }
