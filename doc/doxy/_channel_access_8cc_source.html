76 void ChannelAccess::sendToChannel(cPacket *msg)
77 {
78  const NicEntry::GateList& gateList = cc->getGateList( getParentModule()->getId());
79  NicEntry::GateList::const_iterator i = gateList.begin();
80 
81  if(useSendDirect){
82  // use Andras stuff

84  simtime_t delay = SIMTIME_ZERO;
85  for(; i != --gateList.end(); ++i){
86  //calculate delay (Propagation) to this receiving nic
87  delay = calculatePropagationDelay(i->first);
88 
89  int radioStart = i->second->getId();
90  int radioEnd = radioStart + i->second->size();
91  for (int g = radioStart; g != radioEnd; ++g)
92  sendDirect(static_cast<cPacket*>(msg->dup()),
93  delay, msg->getDuration(), i->second->getOwnerModule(), g);
94  }
95  //calculate delay (Propagation) to this receiving nic
96  delay = calculatePropagationDelay(i->first);
97 
98  int radioStart = i->second->getId();
99  int radioEnd = radioStart + i->second->size();
100  for (int g = radioStart; g != --radioEnd; ++g)
101  sendDirect(static_cast<cPacket*>(msg->dup()),
102  delay, msg->getDuration(), i->second->getOwnerModule(), g);
103 
104  sendDirect(msg, delay, msg->getDuration(), i->second->getOwnerModule(), radioEnd);
105  }
106  else{
107  coreEV << "Nic is not connected to any gates!" << endl;
108  delete msg;
109  }
110  }
111  else{
112  // use our stuff
113  coreEV <<"sendToChannel: sending to gates\n";
114  if( i != gateList.end() ){
115  simtime_t delay = SIMTIME_ZERO;
116  for(; i != --gateList.end(); ++i){
117  //calculate delay (Propagation) to this receiving nic

135 simtime_t ChannelAccess::calculatePropagationDelay(const NicEntry* nic) {
136  if(!usePropagationDelay)
137  return 0;
138 
139  ChannelAccess *const senderModule = this;
140  ChannelAccess *const receiverModule = nic->chAccess;
141  //const simtime_t_cref sStart = simTime();
142 
143  assert(senderModule); assert(receiverModule);
144 
146  Coord sendersPos = senderModule->getMobilityModule()->getCurrentPosition(/*sStart*/);
147  Coord receiverPos = receiverModule->getMobilityModule()->getCurrentPosition(/*sStart*/);
148 
149  // this time-point is used to calculate the distance between sending and receiving host
150  return receiverPos.distance(sendersPos) / BaseWorldUtility::speedOfLight;
151 }
