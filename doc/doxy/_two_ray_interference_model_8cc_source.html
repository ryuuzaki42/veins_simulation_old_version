30 void TwoRayInterferenceModel::filterSignal(AirFrame *frame, const Coord& senderPos, const Coord& receiverPos) {
31  Signal& s = frame->getSignal();
32 
33  const Coord senderPos2D(senderPos.x, senderPos.y);
34  const Coord receiverPos2D(receiverPos.x, receiverPos.y);
35 
36  ASSERT(senderPos.z > 0); // make sure send antenna is above ground
37  ASSERT(receiverPos.z > 0); // make sure receive antenna is above ground
38 
39  double d = senderPos2D.distance(receiverPos2D);
40  double ht = senderPos.z, hr = receiverPos.z;
41 
42  debugEV << "(ht, hr) = (" << ht << ", " << hr << ")" << endl;
43 
44  double d_dir = sqrt( pow (d,2) + pow((ht - hr),2) ); // direct distance
45  double d_ref = sqrt( pow (d,2) + pow((ht + hr),2) ); // distance via ground reflection
46  double sin_theta = (ht + hr)/d_ref;
47  double cos_theta = d/d_ref;
48 
49  double gamma = (sin_theta - sqrt(epsilon_r - pow(cos_theta,2)))/
50  (sin_theta + sqrt(epsilon_r - pow(cos_theta,2)));
51 
52  //is the signal defined to attenuate over frequency?
53  bool hasFrequency = s.getTransmissionPower()->getDimensionSet().hasDimension(Dimension::frequency);
54  debugEV << "Signal contains frequency dimension: " << (hasFrequency ? "yes" : "no") << endl;
55 
56  assert(hasFrequency);
57 
58  debugEV << "Add TwoRayInterferenceModel attenuation (gamma, d, d_dir, d_ref) = (" << gamma << ", " << d << ", " << d_dir << ", " << d_ref << ")" << endl;
59 
60  s.addAttenuation(new TwoRayInterferenceModel::Mapping(gamma, d, d_dir, d_ref, debug));
61 }
