73 void default_veins_TraCI::handleParkingUpdate(cObject* obj) {
74  isParking = traci->getParkingState();
75  if (sendWhileParking == false) {
76  if (isParking == true) {
77  (FindModule<BaseConnectionManager*>::findGlobalModule())->unregisterNic(this->getParentModule()->getSubmodule("nic"));

79  else {
80  Coord pos = traci->getCurrentPosition();
81  (FindModule<BaseConnectionManager*>::findGlobalModule())->registerNic(this->getParentModule()->getSubmodule("nic"), (ChannelAccess*) this->getParentModule()->getSubmodule("nic")->getSubmodule("phy80211p"), &pos);
82  }
