417 
418  assert(signal.getDuration() == frame->getDuration());
419  simtime_t signalEndTime = signal.getReceptionStart() + frame->getDuration();

730 simtime_t BasePhyLayer::setRadioState(int rs) {
731  Enter_Method_Silent();
732  assert(radio);
733 
734  if(txOverTimer && txOverTimer->isScheduled()) {
735  opp_warning("Switched radio while sending an AirFrame. The effects this would have on the transmission are not simulated by the BasePhyLayer!");
736  }
737 
738  simtime_t switchTime = radio->switchTo(rs, simTime());
739 
740  //invalid switch time, we are probably already switching
741  if(switchTime < 0)
742  return switchTime;
743 
744  // if switching is done in exactly zero-time no extra self-message is scheduled
745  if (switchTime == 0.0)
746  {
747  // TODO: in case of zero-time-switch, send no control-message to mac!
748  // maybe call a method finishRadioSwitchingSilent()
749  finishRadioSwitching();
750  } else
751  {
752  sendSelfMessage(radioSwitchingOverTimer, simTime() + switchTime);
753  }
754 
755  return switchTime;
756 }
