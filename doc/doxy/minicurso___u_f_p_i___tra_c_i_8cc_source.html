163 void minicurso_UFPI_TraCI::verifyAndSendCongestionMessage() {
164  if(simTime() - lastSent < 5.0)
165  return;
166  if(speedsList.size() == 0)
167  return;
168  double sum = 0;
169  for(int i = 0; i < speedsList.size(); ++i)
170  sum += speedsList[i].second;
171  double avg = sum/speedsList.size();
172  if(avg < 5.0){
173  string toSend = "";
174  toSend += traci->getRoadId() + "|" + to_string(avg) + "|" + to_string(++minicurso_UFPI_TraCI::messageId);
175  lastSent = simTime();
176  sendMessage(toSend);
177  }
178  double length = traci->commandGetLaneLength(traci->commandGetLaneId());
179  if(length > 0)
180  traci->commandChangeRoute(traci->getRoadId(), avg/length);
181 }
