122 WaveShortMessage* mfcv_rsu::prepareWSM_rsu(std::string name, int lengthBits, t_channel channel, int priority, int rcvId, int serial) {
123  WaveShortMessage* wsm = new WaveShortMessage(name.c_str());
124  wsm->addBitLength(headerLength);

127  switch (channel) {

137  wsm->setSenderPos(curPosition);
138  wsm->setSerial(serial);
139 
140  if (name == "beacon") {

145 
146  //DBG << "\n ttt" << wsm->getSenderSpeed() << " ttt\n";
147 
148  DBG << "Creating Beacon with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
