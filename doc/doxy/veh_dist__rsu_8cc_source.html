78 void vehDist_rsu::messagesReceivedMeasuring(WaveShortMessage* wsm){

87  if ((limitHopCount - wsm->getHopCount()) > it->second. maxHop){
88  it->second.maxHop = (limitHopCount - wsm->getHopCount());
89  }
90  if ((limitHopCount - wsm->getHopCount()) < it->second. minHop){
91  it->second.minHop = (limitHopCount - wsm->getHopCount());
92  }
93  } else {
94  struct messages m;
95  m.HopsCount = 1;
96  m.wsmData = wsm->getWsmData();
97  m.hops = to_string(limitHopCount - wsm->getHopCount());
98  m.maxHop = m.minHop = m.averageHops = limitHopCount - wsm->getHopCount();
99  messagesReceived.insert(make_pair(wsm->getGlobalMessageIdentificaton(), m));
